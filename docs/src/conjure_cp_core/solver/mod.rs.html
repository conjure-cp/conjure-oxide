1//! A high-level API for interacting with constraints solvers.
2//!
3//! This module provides a consistent, solver-independent API for interacting with constraints
4//! solvers. It also provides incremental solving support, and the returning of run stats from
5//! solvers.
6//!
7//! -----
8//!
9//! - [Solver<Adaptor>] provides the API for interacting with constraints solvers.
10//!
11//! - The [SolverAdaptor] trait controls how solving actually occurs and handles translation
12//!   between the [Solver] type and a specific solver.
13//!
14//! - [adaptors] contains all implemented solver adaptors.
15//!
16//! - The [model_modifier] submodule defines types to help with incremental solving / changing a
17//!   model during search. The entrypoint for incremental solving is the [Solver<A,ModelLoaded>::solve_mut]
18//!   function.
19//!
20//! # Examples
21//!
22//! ## A Successful Minion Model
23//!
24//! Note: this example constructs a basic Minion-compatible model instead of using the rewriter.
25//! For a full end-to-end example, see crates/conjure-cp/examples/solver_hello_minion.rs
26//!
27//! ```ignore
28//! use std::sync::{Arc,Mutex};
29//! use conjure_cp_core::parse::get_example_model;
30//! use conjure_cp_core::rule_engine::resolve_rule_sets;
31//! use conjure_cp_core::rule_engine::rewrite_naive;
32//! use conjure_cp_core::solver::{adaptors, Solver, SolverAdaptor};
33//! use conjure_cp_core::solver::states::ModelLoaded;
34//! use conjure_cp_core::Model;
35//! use conjure_cp_core::ast::Domain;
36//! use conjure_cp_core::ast::Declaration;
37//! use conjure_cp_core::solver::SolverFamily;
38//! use conjure_cp_core::context::Context;
39//! use conjure_cp_essence_macros::essence_expr;
40//!
41//! // Define a model for minion.
42//! let context = Context::<'static>::new_ptr_empty(SolverFamily::Minion);
43//! let mut model = Model::new(context);
44//! model.as_submodel_mut().add_symbol(Declaration::new_var("x".into(), Domain::Bool));
45//! model.as_submodel_mut().add_symbol(Declaration::new_var("y".into(), Domain::Bool));
46//! model.as_submodel_mut().add_constraint(essence_expr!{x != y});
47//!
48//! // Solve using Minion.
49//! let solver = Solver::new(adaptors::Minion::new());
50//! let solver: Solver<adaptors::Minion,ModelLoaded> = solver.load_model(model).unwrap();
51//!
52//! // In this example, we will count solutions.
53//! //
54//! // The solver interface is designed to allow adaptors to use multiple-threads / processes if
55//! // necessary. Therefore, the callback type requires all variables inside it to have a static
56//! // lifetime and to implement Send (i.e. the variable can be safely shared between theads).
57//! //
58//! // We use Arc<Mutex<T>> to create multiple references to a threadsafe mutable
59//! // variable of type T.
60//! //
61//! // Using the move |x| ... closure syntax, we move one of these references into the closure.
62//! // Note that a normal closure borrow variables from the parent so is not
63//! // thread-safe.
64//!
65//! let counter_ref = Arc::new(Mutex::new(0));
66//! let counter_ref_2 = counter_ref.clone();
67//! solver.solve(Box::new(move |_| {
68//!   let mut counter = (*counter_ref_2).lock().unwrap();
69//!   *counter += 1;
70//!   true
71//!   }));
72//!
73//! let mut counter = (*counter_ref).lock().unwrap();
74//! assert_eq!(*counter,2);
75//! ```
76//!
77//! # The Solver callback function
78//!
79//! The callback function given to `solve` is called whenever a solution is found by the solver.
80//!
81//! Its return value can be used to control how many solutions the solver finds:
82//!
83//! * If the callback function returns `true`, solver execution continues.
84//! * If the callback function returns `false`, the solver is terminated.
85//!

88//
89// Solver interfaces can only be implemented inside this module, due to the SolverAdaptor crate
90// being sealed.
91//

96// NotModifiable.

164/// A common interface for calling underlying solver APIs inside a [`Solver`].
165///
166/// Implementations of this trait aren't directly callable and should be used through [`Solver`] .
167///
168/// The below documentation lists the formal requirements that all implementations of
169/// [`SolverAdaptor`] should follow - **see the top level module documentation and [`Solver`] for
170/// usage details.**
171///
172/// # Encapsulation
173///
174///  The [`SolverAdaptor`] trait **must** only be implemented inside a submodule of this one,
175///  and **should** only be called through [`Solver`].
176///
177/// The `private::Sealed` trait and `private::Internal` type enforce these requirements by only
178/// allowing trait implementations and calling of methods of SolverAdaptor to occur inside this
179/// module.
180///
181/// # Thread Safety
182///
183/// Multiple instances of [`Solver`] can be run in parallel across multiple threads.
184///
185/// [`Solver`] provides no concurrency control or thread-safety; therefore, adaptors **must**
186/// ensure that multiple instances of themselves can be ran in parallel. This applies to all
187/// stages of solving including having two active `solve()` calls happening at a time, loading
188/// a model while another is mid-solve, loading two models at once, etc.
189///
190/// A [SolverAdaptor] **may** use whatever threading or process model it likes underneath the hood,
191/// as long as it obeys the above.
192///
193/// Method calls **should** block instead of erroring where possible.
194///
195/// Underlying solvers that only have one instance per process (such as Minion) **should** block
196/// (eg. using a [`Mutex<()>`](`std::sync::Mutex`)) to run calls to
197/// [`Solver<A,ModelLoaded>::solve()`] and [`Solver<A,ModelLoaded>::solve_mut()`] sequentially.
198pub trait SolverAdaptor: private::Sealed + Any {
199    /// Runs the solver on the given model.
200    ///
201    /// Implementations of this function **must** call the user provided callback whenever a solution
202    /// is found. If the user callback returns `true`, search should continue, if the user callback
203    /// returns `false`, search should terminate.
204    ///
205    /// # Returns
206    ///
207    /// If the solver terminates without crashing a [SolveSuccess] struct **must** returned. The
208    /// value of [SearchStatus] can be used to denote whether the underlying solver completed its
209    /// search or not. The latter case covers most non-crashing "failure" cases including user
210    /// termination, timeouts, etc.
211    ///
212    /// To help populate [SearchStatus], it may be helpful to implement counters that track if the
213    /// user callback has been called yet, and its return value. This information makes it is
214    /// possible to distinguish between the most common search statuses:
215    /// [SearchComplete::HasSolutions], [SearchComplete::NoSolutions], and
216    /// [SearchIncomplete::UserTerminated].
217    fn solve(
218        &mut self,
219        callback: SolverCallback,
220        _: private::Internal,
221    ) -> Result<SolveSuccess, SolverError>;
222
223    /// Runs the solver on the given model, allowing modification of the model through a
224    /// [`ModelModifier`].
225    ///
226    /// Implementations of this function **must** return [`OpNotSupported`](`ModificationFailure::OpNotSupported`)
227    /// if modifying the model mid-search is not supported.
228    ///
229    /// Otherwise, this should work in the same way as [`solve`](SolverAdaptor::solve).
230    fn solve_mut(
231        &mut self,
232        callback: SolverMutCallback,
233        _: private::Internal,
234    ) -> Result<SolveSuccess, SolverError>;
235    fn load_model(&mut self, model: Model, _: private::Internal) -> Result<(), SolverError>;
236    fn init_solver(&mut self, _: private::Internal) {}
237
238    /// Get the solver family that this solver adaptor belongs to
239    fn get_family(&self) -> SolverFamily;
240
241    /// Gets the name of the solver adaptor for pretty printing.
242    fn get_name(&self) -> Option<String> {
243        None
244    }
245
246    /// Adds the solver adaptor name and family (if they exist) to the given stats object.
247    fn add_adaptor_info_to_stats(&self, stats: SolverStats) -> SolverStats {
248        SolverStats {
249            solver_adaptor: self.get_name(),
250            solver_family: Some(self.get_family()),
251            ..stats
252        }
253    }
254
255    /// Writes a solver input file to the given writer.
256    ///
257    /// This method is for debugging use only, and there are no plans to make the solutions
258    /// obtained by running this file through the solver translatable back into high-level Essence.
259    ///
260    /// This file is runnable using the solvers command line interface. E.g. for Minion, this
261    /// outputs a valid .minion file.
262    ///
263    ///
264    /// # Implementation
265    /// + It can be helpful for this file to contain comments linking constraints and variables to
266    ///   their original essence, but this is not required.
267    ///
268    /// + This function is ran after model loading but before solving - therefore, it is safe for
269    ///   solving to mutate the model object.
270    fn write_solver_input_file(&self, writer: &mut impl Write) -> Result<(), std::io::Error>;
271}

273/// An abstract representation of a constraints solver.
274///
275/// [Solver] provides a common interface for interacting with a constraint solver. It also
276/// abstracts over solver-specific datatypes, handling the translation to/from [conjure_cp_core::ast]
277/// types for a model and its solutions.
278///
279/// Details of how a model is solved is specified by the [SolverAdaptor]. This includes: the
280/// underlying solver used, the translation of the model to a solver compatible form, how solutions
281/// are translated back to [conjure_cp_core::ast] types, and how incremental solving is implemented.
282/// As such, there may be multiple [SolverAdaptor] implementations for a single underlying solver:
283/// e.g. one adaptor may give solutions in a representation close to the solvers, while another may
284/// attempt to rewrite it back into Essence.
285///
286#[derive(Clone)]
287pub struct Solver<A: SolverAdaptor, State: SolverState = Init> {
288    state: State,
289    adaptor: A,
290    context: Option<Arc<RwLock<Context<'static>>>>,
291}
292
293impl<Adaptor: SolverAdaptor> Solver<Adaptor> {
294    pub fn new(solver_adaptor: Adaptor) -> Solver<Adaptor> {
295        let mut solver = Solver {
296            state: Init,
297            adaptor: solver_adaptor,
298            context: None,
299        };
300
301        solver.adaptor.init_solver(private::Internal);
302        solver
303    }
304
305    pub fn get_family(&self) -> SolverFamily {
306        self.adaptor.get_family()
307    }
308}

310impl<A: SolverAdaptor> Solver<A, Init> {
311    pub fn load_model(mut self, model: Model) -> Result<Solver<A, ModelLoaded>, SolverError> {
312        let solver_model = &mut self.adaptor.load_model(model.clone(), private::Internal)?;
313        Ok(Solver {
314            state: ModelLoaded,
315            adaptor: self.adaptor,
316            context: Some(model.context.clone()),
317        })
318    }
319}

321impl<A: SolverAdaptor> Solver<A, ModelLoaded> {
322    pub fn solve(
323        mut self,
324        callback: SolverCallback,
325    ) -> Result<Solver<A, ExecutionSuccess>, SolverError> {
326        #[allow(clippy::unwrap_used)]
327        let start_time = Instant::now();
328
329        #[allow(clippy::unwrap_used)]
330        let result = self.adaptor.solve(callback, private::Internal);
331
332        let duration = start_time.elapsed();
333
334        match result {
335            Ok(x) => {
336                let stats = self
337                    .adaptor
338                    .add_adaptor_info_to_stats(x.stats)
339                    .with_timings(duration.as_secs_f64());
340
341                Ok(Solver {
342                    adaptor: self.adaptor,
343                    state: ExecutionSuccess {
344                        stats,
345                        status: x.status,
346                        _sealed: private::Internal,
347                    },
348                    context: self.context,
349                })
350            }
351            Err(x) => Err(x),
352        }
353    }
354
355    pub fn solve_mut(
356        mut self,
357        callback: SolverMutCallback,
358    ) -> Result<Solver<A, ExecutionSuccess>, SolverError> {
359        #[allow(clippy::unwrap_used)]
360        let start_time = Instant::now();
361
362        #[allow(clippy::unwrap_used)]
363        let result = self.adaptor.solve_mut(callback, private::Internal);
364
365        let duration = start_time.elapsed();
366
367        match result {
368            Ok(x) => {
369                let stats = self
370                    .adaptor
371                    .add_adaptor_info_to_stats(x.stats)
372                    .with_timings(duration.as_secs_f64());
373
374                Ok(Solver {
375                    adaptor: self.adaptor,
376                    state: ExecutionSuccess {
377                        stats,
378                        status: x.status,
379                        _sealed: private::Internal,
380                    },
381                    context: self.context,
382                })
383            }
384            Err(x) => Err(x),
385        }
386    }
387
388    /// Writes a solver input file to the given writer.
389    ///
390    /// This method is for debugging use only, and there are no plans to make the solutions
391    /// obtained by running this file through the solver translatable back into high-level Essence.
392    ///
393    /// This file is runnable using the solvers command line interface. E.g. for Minion, this
394    /// outputs a valid .minion file.
395    ///
396    /// This function is only available in the `ModelLoaded` state as solvers are allowed to edit
397    /// the model in place.
398    pub fn write_solver_input_file(&self, writer: &mut impl Write) -> Result<(), std::io::Error> {
399        self.adaptor.write_solver_input_file(writer)
400    }
401}

403impl<A: SolverAdaptor> Solver<A, ExecutionSuccess> {
404    pub fn stats(&self) -> SolverStats {
405        self.state.stats.clone()
406    }

408    // Saves this solvers stats to the global context as a "solver run"
409    pub fn save_stats_to_context(&self) {
410        #[allow(clippy::unwrap_used)]
411        #[allow(clippy::expect_used)]
412        self.context
413            .as_ref()
414            .expect("")
415            .write()
416            .unwrap()
417            .stats
418            .add_solver_run(self.stats());
419    }

422        self.stats().conjure_solver_wall_time_s
423    }

426/// Errors returned by [Solver] on failure.
427#[non_exhaustive]
428#[derive(Debug, Error, Clone)]
429pub enum SolverError {
430    #[error("operation not implemented yet: {0}")]
431    OpNotImplemented(String),
432
433    #[error("operation not supported: {0}")]
434    OpNotSupported(String),
435
436    #[error("model feature not supported: {0}")]
437    ModelFeatureNotSupported(String),
438
439    #[error("model feature not implemented yet: {0}")]
440    ModelFeatureNotImplemented(String),
441
442    // use for semantics / type errors, use the above for syntax
443    #[error("model invalid: {0}")]
444    ModelInvalid(String),
445
446    #[error("error during solver execution: not implemented: {0}")]
447    RuntimeNotImplemented(String),
448
449    #[error("error during solver execution: {0}")]
450    Runtime(String),
451}

459pub enum SearchStatus {
460    /// The search was complete (i.e. the solver found all possible solutions)
461    Complete(SearchComplete),
462    /// The search was incomplete (i.e. it was terminated before all solutions were found)
463    Incomplete(SearchIncomplete),
464}
