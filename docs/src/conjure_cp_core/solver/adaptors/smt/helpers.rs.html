10/// Use 32-bit 2's complement signed bit-vectors
11pub const BV_SIZE: u32 = 32;
12
13/// A function which encodes a restriction for a specific variable. Given an AST of the correct
14/// sort, constructs a boolean assertion which will ensure the variable has the correct domain.
15type RestrictFn = Box<dyn Fn(&Dynamic) -> Bool>;
16
17/// Returns the Oxide domain as a Z3 sort, along with a function to restrict a variable of that sort
18/// to the original domain's restrictions.
19pub fn domain_to_sort(
20    domain: &GroundDomain,
21    theories: &TheoryConfig,
22) -> SolverResult<(Sort, RestrictFn)> {
23    use IntTheory::{Bv, Lia};
24
25    match (theories.ints, domain) {
26        // Booleans of course have the same domain in SMT, so no restriction required
27        (_, GroundDomain::Bool) => Ok((Sort::bool(), Box::new(|_| Bool::from_bool(true)))),
28
29        // Return a disjunction of the restrictions each range of the domain enforces
30        // I.e. `x: int(1, 3..5)` -> `or([x = 1, x >= 3 /\ x <= 5])`
31        (Lia, GroundDomain::Int(ranges)) => {
32            let ranges = ranges.clone();
33            let restrict_fn = move |ast: &Dynamic| {
34                let int = ast.as_int().unwrap();
35                let restrictions: Vec<_> = ranges
36                    .iter()
37                    .map(|range| int_range_to_int_restriction(&int, range))
38                    .collect();
39                Bool::or(restrictions.as_slice())
40            };
41            Ok((Sort::int(), Box::new(restrict_fn)))
42        }
43        (Bv, GroundDomain::Int(ranges)) => {
44            let ranges = ranges.clone();
45            let restrict_fn = move |ast: &Dynamic| {
46                let bv = ast.as_bv().unwrap();
47                let restrictions: Vec<_> = ranges
48                    .iter()
49                    .map(|range| int_range_to_bv_restriction(&bv, range))
50                    .collect();
51                Bool::or(restrictions.as_slice())
52            };
53            Ok((Sort::bitvector(BV_SIZE), Box::new(restrict_fn)))
54        }
55
56        (_, GroundDomain::Matrix(val_domain, idx_domains)) => {
57            // We constrain the inner values of the domain recursively
58            // I.e. every way to index the array must give a value in the correct domain
59
60            let (range_sort, restrict_val) = match idx_domains.as_slice() {
61                [_] => domain_to_sort(val_domain, theories),
62                [_, tail @ ..] => {
63                    // Treat as a matrix containing (n-1)-dimensional matrices
64                    let inner_domain = GroundDomain::Matrix(val_domain.clone(), tail.to_vec());
65                    domain_to_sort(&inner_domain, theories)
66                }
67                [] => Err(SolverError::ModelInvalid(
68                    "empty matrix index domain".into(),
69                )),
70            }?;
71
72            // No need to constrain the indices themselves, that's done through SafeIndex/InDomain
73            let idx_domain = &idx_domains[0];
74            let (domain_sort, _) = domain_to_sort(idx_domain.as_ref(), theories)?;
75
76            // Use the lower dimension's restricting fn to restrict all indexes in this dimension
77            let idx_asts = domain_to_ast_vec(theories, idx_domain.as_ref())?;
78            let restrict_fn = move |ast: &Dynamic| {
79                let arr = ast.as_array().unwrap();
80                let restrictions: Vec<_> = idx_asts
81                    .iter()
82                    .map(|idx_ast| (restrict_val)(&arr.select(idx_ast)))
83                    .collect();
84                Bool::and(restrictions.as_slice())
85            };
86            Ok((
87                Sort::array(&domain_sort, &range_sort),
88                Box::new(restrict_fn),
89            ))
90        }
91
92        (_, GroundDomain::Set(attr, elem_domain)) => {
93            let (val_sort, _) = domain_to_sort(elem_domain, theories)?;
94
95            // Restrict the size of the set
96            let member_asts = domain_to_ast_vec(theories, elem_domain)?;
97            let attr_size = attr.size.clone();
98            let restrict_fn = move |ast: &Dynamic| {
99                let set = ast.as_set().unwrap();
100                let is_member: Vec<_> = member_asts
101                    .iter()
102                    .map(|val| set.member(val).ite(&Int::from(1), &Int::from(0)))
103                    .collect();
104                let size = Int::add(&is_member);
105                match attr_size {
106                    Range::Single(n) => size.eq(Int::from(n)),
107                    Range::UnboundedL(r) => size.le(Int::from(r)),
108                    Range::UnboundedR(l) => size.ge(Int::from(l)),
109                    Range::Bounded(l, r) => {
110                        Bool::and(&[size.ge(Int::from(l)), size.le(Int::from(r))])
111                    }
112                    Range::Unbounded => Bool::from_bool(true),
113                }
114            };
115
116            Ok((Sort::set(&val_sort), Box::new(restrict_fn)))
117        }
118
119        _ => Err(SolverError::ModelFeatureNotImplemented(format!(
120            "sort for '{domain}' not implemented"
121        ))),
122    }
123}

125/// Returns a domain as a vector of Z3 AST literals.
126pub fn domain_to_ast_vec(
127    theory_config: &TheoryConfig,
128    domain: &GroundDomain,
129) -> SolverResult<Vec<Dynamic>> {
130    let lits = domain
131        .values()
132        .map_err(|err| SolverError::Runtime(err.to_string()))?;
133    lits.map(|lit| literal_to_ast(theory_config, &lit))
134        .collect()
135}

137/// Returns a boolean expression restricting the given integer variable to the given range.
138pub fn int_range_to_int_restriction(var: &Int, range: &Range<i32>) -> Bool {
139    match range {
140        Range::Single(n) => var.eq(Int::from(*n)),
141        Range::UnboundedL(r) => var.le(Int::from(*r)),
142        Range::UnboundedR(l) => var.ge(Int::from(*l)),
143        Range::Bounded(l, r) => Bool::and(&[var.ge(Int::from(*l)), var.le(Int::from(*r))]),
144        _ => bug!("int ranges should not be unbounded"),
145    }
146}
147
148/// Returns a boolean expression restricting the given bitvector variable to the given integer range.
149pub fn int_range_to_bv_restriction(var: &BV, range: &Range<i32>) -> Bool {
150    match range {
151        Range::Single(n) => var.eq(BV::from_i64(*n as i64, BV_SIZE)),
152        Range::UnboundedL(r) => var.bvsle(BV::from_i64(*r as i64, BV_SIZE)),
153        Range::UnboundedR(l) => var.bvsge(BV::from_i64(*l as i64, BV_SIZE)),
154        Range::Bounded(l, r) => Bool::and(&[
155            var.bvsge(BV::from_i64(*l as i64, BV_SIZE)),
156            var.bvsle(BV::from_i64(*r as i64, BV_SIZE)),
157        ]),
158        _ => bug!("int ranges should not be unbounded"),
159    }
160}

162pub fn name_to_symbol(name: &Name) -> SolverResult<Symbol> {
163    match name {
164        Name::User(ustr) => Ok(Symbol::String((*ustr).into())),
165        Name::Machine(num) => Ok(Symbol::Int(*num as u32)),
166        Name::Represented(parts) => {
167            let (name, rule_str, suffix) = parts.as_ref();
168            Ok(Symbol::String(format!("{name}#{rule_str}_{suffix}")))
169        }
170        _ => Err(SolverError::ModelFeatureNotImplemented(format!(
171            "variable '{name}' name is unsupported"
172        ))),
173    }
174}

176/// Converts an atom (literal or reference) into an AST node.
177pub fn atom_to_ast(
178    theory_config: &TheoryConfig,
179    store: &SymbolStore,
180    atom: &Atom,
181) -> SolverResult<Dynamic> {
182    match atom {
183        Atom::Reference(decl) => store
184            .get(&decl.name())
185            .ok_or(SolverError::ModelInvalid(format!(
186                "variable '{}' does not exist",
187                decl.name()
188            )))
189            .map(|(_, ast, _)| ast)
190            .cloned(),
191        Atom::Literal(lit) => literal_to_ast(theory_config, lit),
192        _ => Err(SolverError::ModelFeatureNotImplemented(format!(
193            "atom sort not implemented: {atom}"
194        ))),
195    }
196}

198/// Converts a CO literal (expression containing no variables) into an AST node.
199pub fn literal_to_ast(theory_config: &TheoryConfig, lit: &Literal) -> SolverResult<Dynamic> {
200    match lit {
201        Literal::Bool(b) => Ok(Bool::from_bool(*b).into()),
202        Literal::Int(n) => Ok(match theory_config.ints {
203            IntTheory::Lia => Int::from(*n).into(),
204            IntTheory::Bv => BV::from_i64(*n as i64, BV_SIZE).into(),
205        }),
206        _ => Err(SolverError::ModelFeatureNotImplemented(format!(
207            "literal type not implemented: {lit}"
208        ))),
209    }
