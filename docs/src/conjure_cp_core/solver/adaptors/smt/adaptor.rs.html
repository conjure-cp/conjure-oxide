59impl SolverAdaptor for Smt {
60    fn solve(
61        &mut self,
62        callback: SolverCallback,
63        _: private::Internal,
64    ) -> Result<SolveSuccess, SolverError> {
65        let solver_send = self.solver_inst.synchronized();
66        let store_send = self.store.synchronized();
67        let mut stats: SolverStats = Default::default();
68
69        // Apply config when getting solutions
70        let (search_complete, final_z3_time) = with_z3_config(&self.solver_cfg, move || {
71            let solver = solver_send.recover();
72            let mut final_z3_time: Option<f64> = None;
73
74            let solutions = solver
75                .into_solutions_with_statistics(store_send.recover(), true)
76                .take_while(|(store, z3_stats)| {
77                    let time = z3_stats.value("time");
78                    if let Some(z3::StatisticsValue::Double(time)) = time {
79                        final_z3_time = Some(time);
80                    }
81                    (callback)(store.as_literals_map().unwrap())
82                });
83
84            // Consume iterator and get whether there are solutions
85            let search_complete = match solutions.count() {
86                0 => SearchComplete::NoSolutions,
87                _ => SearchComplete::HasSolutions,
88            };
89            (search_complete, final_z3_time)
90        });
91
92        if let Some(time) = final_z3_time {
93            stats.solver_time_s = time;
94        }
95
96        Ok(SolveSuccess {
97            stats,
98            status: SearchStatus::Complete(search_complete),
99        })
100    }

102    fn solve_mut(
103        &mut self,
104        callback: SolverMutCallback,
105        _: private::Internal,

117            submodel.constraints().as_slice(),
118        )?;
119        Ok(())
120    }

136    }

147impl IntoSolutionsWithStatistics for z3::Solver {
148    fn into_solutions_with_statistics<T: Solvable>(
149        self,
150        t: T,
151        model_completion: bool,
152    ) -> impl FusedIterator<Item = (T::ModelInstance, Statistics)> {

162struct SolverStatsIterator<T> {
163    solver: Solver,
164    ast: T,
165    model_completion: bool,
166}
167
168// copy-pasted from upstream except this runs get_statistics
169impl<T: Solvable> Iterator for SolverStatsIterator<T> {
170    type Item = (T::ModelInstance, Statistics);
171
172    fn next(&mut self) -> Option<Self::Item> {
173        match self.solver.check() {
174            SatResult::Sat => {
175                // right after we solve, before we generate the model grab statistics
176                let stats = self.solver.get_statistics();
177                let model = self.solver.get_model()?;
178                let instance = self.ast.read_from_model(&model, self.model_completion)?;
179                let counterexample = self.ast.generate_constraint(&instance);
180                self.solver.assert(counterexample);
181                // and return them through the iterator
182                Some((instance, stats))
183            }
184            _ => None,
185        }
186    }
