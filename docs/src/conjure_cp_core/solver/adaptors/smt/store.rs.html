14/// Maps CO variable names to their CO domains, Z3 symbolic constants, and Z3 symbols.
15#[derive(Clone, Debug)]
16pub struct SymbolStore {
17    map: HashMap<Name, (Moo<GroundDomain>, Dynamic, Symbol)>,
18    theories: TheoryConfig,
19}
20
21impl SymbolStore {
22    pub fn new(theories: TheoryConfig) -> Self {
23        SymbolStore {
24            map: HashMap::new(),
25            theories,
26        }
27    }
28
29    pub fn insert(
30        &mut self,
31        name: Name,
32        val: (Moo<GroundDomain>, Dynamic, Symbol),
33    ) -> Option<(Moo<GroundDomain>, Dynamic, Symbol)> {
34        self.map.insert(name, val)
35    }
36
37    pub fn get(&self, name: &Name) -> Option<&(Moo<GroundDomain>, Dynamic, Symbol)> {
38        self.map.get(name)
39    }
40}

42impl Solvable for SymbolStore {
43    type ModelInstance = LiteralStore;
44
45    fn read_from_model(
46        &self,
47        model: &Model,
48        model_completion: bool,
49    ) -> Option<Self::ModelInstance> {
50        let mut new_store = LiteralStore::new();
51        for (name, (domain, ast, sym)) in self.map.iter() {
52            // Get the interpretation of each constant
53            let (ast, lit) =
54                interpret(model, (domain, ast), model_completion, &self.theories).unwrap();
55            new_store.map.insert(name.clone(), (ast, lit));
56        }
57        Some(new_store)
58    }
59
60    fn generate_constraint(&self, model: &Self::ModelInstance) -> Bool {
61        let bools: Vec<_> = self
62            .map
63            .iter()
64            .map(|(name, (domain, ast, _))| {
65                let (other, _) = model.map.get(name).unwrap();
66                match domain.as_ref() {
67                    GroundDomain::Matrix(_, idx_domains) => {
68                        // Rather than just setting `array != other_array`, we need to do it for every element
69                        // Otherwise, Z3 can generate new arrays which are equal over the domain
70                        // but different otherwise (and this loops infinitely)
71
72                        let neqs: Vec<_> = idx_domains
73                            .iter()
74                            .map(|domain| domain_to_ast_vec(&self.theories, domain).unwrap())
75                            .multi_cartesian_product()
76                            .map(|idxs| {
77                                idxs.iter()
78                                    .fold((ast.clone(), other.clone()), |(a, b), idx| {
79                                        (
80                                            a.as_array().unwrap().select(idx),
81                                            b.as_array().unwrap().select(idx),
82                                        )
83                                    })
84                            })
85                            .map(|(a, b)| a.ne(b))
86                            .collect();
87                        Bool::or(&neqs)
88                    }
89                    GroundDomain::Set(_, elem_domain) => {
90                        let set = ast.as_set().unwrap();
91                        let other_set = other.as_set().unwrap();
92                        let neqs: Vec<_> = domain_to_ast_vec(&self.theories, elem_domain)
93                            .unwrap()
94                            .iter()
95                            .map(|ast| set.member(ast).ne(other_set.member(ast)))
96                            .collect();
97                        Bool::or(&neqs)
98                    }
99
100                    // Any other variables are just directly compared
101                    _ => ast.ne(other),
102                }
103            })
104            .collect();
105        Bool::or(bools.as_slice())
106    }
107}

109unsafe impl Translate for SymbolStore {
110    fn translate(&self, ctx: &Context) -> Self {
111        let mut new_map = HashMap::new();
112        for (name, (domain, ast, sym)) in self.map.iter() {
113            let new_ast = ast.translate(ctx);
114            let new_sym = sym.clone(); // Symbols are not translated
115            new_map.insert(name.clone(), (domain.clone(), new_ast, new_sym));
116        }
117        SymbolStore {
118            map: new_map,
119            theories: self.theories,
120        }
121    }
122}

124/// Maps CO variable names to their literal values in both the Z3 model and Essence.

126pub struct LiteralStore {

128}
129
130impl LiteralStore {

136
137    /// Return this store as a mapping of CO names to literals
138    pub fn as_literals_map(&self) -> SolverResult<HashMap<Name, Literal>> {
139        let iter = self
140            .map
141            .iter()
142            .map(|(name, (_, lit))| (name.clone(), lit.clone()));
143        Ok(HashMap::from_iter(iter))
144    }

162/// Interprets the given value within the given model and returns a CO literal.
163/// The value can be any AST including a symbolic constant, as long as it is defined in the model.
164///
165/// This method makes any interpretations necessary to fully evaluate the literal. E.g. for arrays
166/// it must enumerate over all elements in the index domain and evaluate the elements as literals.
167fn interpret(
168    model: &Model,
169    value: (&Moo<GroundDomain>, &Dynamic),
170    model_completion: bool,
171    theories: &TheoryConfig,
172) -> SolverResult<(Dynamic, Literal)> {
173    use IntTheory::{Bv, Lia};
174
175    let (domain, var_ast) = value;
176    let lit_ast = model
177        .eval(var_ast, model_completion)
178        .ok_or(SolverError::Runtime(format!(
179            "could not interpret variable: {var_ast}"
180        )))?;
181
182    let literal = match (theories.ints, domain.as_ref()) {
183        (_, GroundDomain::Bool) => {
184            let bool_ast = lit_ast.as_bool().unwrap();
185            let bool = bool_ast.as_bool().unwrap();
186            Ok(Literal::Bool(bool))
187        }
188        (Lia, GroundDomain::Int(_)) => {
189            let int_ast = lit_ast.as_int().unwrap();
190            let int = int_ast
191                .as_i64()
192                .ok_or(SolverError::Runtime(format!(
193                    "could not cast to i64: {lit_ast}"
194                )))?
195                .try_into()
196                .map_err(|err| {
197                    SolverError::Runtime(format!("value {lit_ast} out of range: {err}"))
198                })?;
199            Ok(Literal::Int(int))
200        }
201        (Bv, GroundDomain::Int(_)) => {
202            // BVs do not sign-extend when returning u64s (if they are < 64 bits)
203            // To correctly retrieve negative numbers, we downsize to a u32 and then bit-wise
204            // interpret it as an i32, rather than casting.
205            // See https://github.com/prove-rs/z3.rs/issues/458
206            let bv_ast = lit_ast.as_bv().unwrap();
207            let unsigned_64: u64 = bv_ast.as_u64().ok_or(SolverError::Runtime(format!(
208                "could not retrieve u64: {lit_ast}"
209            )))?;
210            let unsigned_32: u32 = unsigned_64.try_into().map_err(|err| {
211                SolverError::Runtime(format!("value {lit_ast} out of range: {err}"))
212            })?;
213            let signed = i32::from_ne_bytes(unsigned_32.to_ne_bytes());
214            Ok(Literal::Int(signed))
215        }
216        (_, GroundDomain::Matrix(val_domain, idx_domains)) => {
217            let arr_ast = lit_ast.as_array().unwrap();
218
219            let inner_domain = match idx_domains.as_slice() {
220                [idx_domain] => val_domain.clone(),
221                [idx_domain, tail @ ..] => {
222                    Moo::new(GroundDomain::Matrix(val_domain.clone(), tail.to_vec()))
223                }
224                [] => return Err(SolverError::Runtime("empty matrix index domain".into())),
225            };
226
227            let indices = domain_to_ast_vec(theories, &idx_domains[0])?;
228            let elements_res: Result<Vec<_>, _> = indices
229                .iter()
230                .map(|idx| model.eval(&arr_ast.select(idx), model_completion).unwrap())
231                .map(|ast| interpret(model, (&inner_domain, &ast), model_completion, theories))
232                .map(|res| res.map(|(_, lit)| lit))
233                .collect();
234            let elements = elements_res?;
235
236            Ok(Literal::AbstractLiteral(AbstractLiteral::Matrix(
237                elements,
238                domain.clone(),
239            )))
240        }
241        (_, GroundDomain::Set(_, val_domain)) => {
242            let set_ast = lit_ast.as_set().unwrap();
243
244            // Collect every member of the domain that is in the set
245            let dom_iter = val_domain
246                .values()
247                .map_err(|_| SolverError::Runtime("could not construct domain iterator".into()))?;
248            let members: Result<Vec<Literal>, SolverError> = dom_iter
249                .map(|lit| {
250                    // We want to bubble errors up but also only collect set members
251                    let lit_ast = literal_to_ast(theories, &lit)?;
252                    let is_member = model
253                        .eval(&set_ast.member(&lit_ast), model_completion)
254                        .unwrap()
255                        .as_bool()
256                        .unwrap();
257                    Ok(is_member.then_some(lit))
258                })
259                .filter_map(|res: Result<Option<Literal>, SolverError>| res.ok().flatten().map(Ok))
260                .collect();
261            let members = members?;
262
263            Ok(Literal::AbstractLiteral(AbstractLiteral::Set(members)))
264        }
265        _ => Err(SolverError::RuntimeNotImplemented(format!(
266            "conversion from AST to literal of type '{domain}' not implemented: {lit_ast}"
267        ))),
268    }?;
269    Ok((lit_ast, literal))
