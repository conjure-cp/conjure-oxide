35impl<'a> Context<'a> {

39        rules: Vec<RuleData<'a>>,
40        rule_sets: Vec<&'a RuleSet<'a>>,

53impl Context<'static> {

78impl Debug for Context<'_> {
79    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
80        let target_solver_family: Option<SolverFamily> = self.target_solver_family;
81        let extra_rule_set_names: Vec<String> = self.extra_rule_set_names.clone();
82        let rules: Vec<&str> = self.rules.iter().map(|rd| rd.rule.name).collect();
83        let rule_sets: Vec<&str> = self.rule_sets.iter().map(|r| r.name).collect();
84
85        write!(
86            f,
87            "Context {{\n\
88            \ttarget_solver_family: {target_solver_family:?}\n\
89            \textra_rule_set_names: {extra_rule_set_names:?}\n\
90            \trules: {rules:?}\n\
91            \trule_sets: {rule_sets:?}\n\
92        }}"
93        )
94    }
