24/// Creates a [`Domain::Int`](ast::Domain::Int).
25///
26/// # Examples
27/// ```
28/// use conjure_cp_core::{domain_int,range,ast::Domain};
29///
30/// let a = 2*10;
31/// assert_eq!(domain_int!(1..5,a+2,), Domain::Int(vec![range!(1..5),range!(a+2)]));
32/// assert_eq!(domain_int!(), Domain::Int(vec![]));
33/// ```
34#[macro_export]
35macro_rules! domain_int {
36    () => {$crate::ast::Domain::Int(vec![])};
37
38    // when parsing expressions, rust groups 1..2 into a single token tree, (1..2)
39    // however, we want it to be three seperate token trees [1,..,2] for parsing.
40    // use defile to turn it back into 3 token trees
41    ($($e:expr),+ $(,)?) => {::defile::defile! { $crate::ast::Domain::Int(vec![$($crate::range!(@$e)),+]) } };
42}

44/// Creates a [`Range`](ast::Range).
45///
46/// # Examples
47///
48/// ```
49/// use conjure_cp_core::{range,ast::Range};
50///
51/// let a = 2*10;
52/// assert_eq!(range!(..a),Range::UnboundedL(a));
53/// assert_eq!(range!(2*5..),Range::UnboundedR(10));
54/// assert_eq!(range!(..10),Range::UnboundedL(10));
55/// assert_eq!(range!(1..5),Range::Bounded(1,5));
56/// assert_eq!(range!(Some(10).unwrap()),Range::Single(10));
57/// ```
58#[macro_export]
59macro_rules! range {
60    // decl macros have no lookahead, hence this nonsense with pushdown automata.
61
62    // @hasLowerbound: have atleast one token of the lower bound on the stack
63    (@hasLowerBound [$($lower:tt)+] -> ..) => {$crate::ast::Range::UnboundedR($crate::as_expr!($($lower)+))};
64    (@hasLowerBound [$($lower:tt)+] -> .. $($tail:tt)+) => {$crate::ast::Range::Bounded($crate::as_expr!($($lower)+),$crate::as_expr!($($tail)+))};
65    (@hasLowerBound [$($lower:tt)+] -> $b:tt $($tail:tt)*) => {range!(@hasLowerBound [$($lower)+ $b] -> $($tail)*)};
66    (@hasLowerBound [$($lower:tt)+] ->) => {$crate::ast::Range::Single($crate::as_expr!($($lower)+))};
67
68    // initial tokens
69    (.. $($a:tt)+) => {$crate::ast::Range::UnboundedL($crate::as_expr!($($a)+))};
70
71    ($a:tt $($tail:tt)*) => {range!(@hasLowerBound [$a] -> $($tail)*)};
72
73}
