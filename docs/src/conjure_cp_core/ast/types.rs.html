1use itertools::Itertools;
2use polyquine::Quine;
3use serde::{Deserialize, Serialize};
4use std::fmt::{Display, Formatter};
5
6#[derive(Clone, Debug, PartialEq, Eq, Deserialize, Serialize, Hash, Quine)]
7pub enum ReturnType {
8    Int,
9    Bool,
10    Matrix(Box<ReturnType>),
11    Set(Box<ReturnType>),
12    Tuple(Vec<ReturnType>),
13    Record(Vec<ReturnType>),
14    Function(Box<ReturnType>, Box<ReturnType>),
15
16    /// An unknown type
17    ///
18    /// This can be found inside the types of empty abstract literals.
19    ///
20    /// To understand why, consider the typing of a set literal.  We construct the type of a set
21    /// literal by looking at the type of its items (e.g. {1,2,3} is type `set(int)`, as 1 is an
22    /// int). However, if it has no items, we can't do this, so we give it the type `set(unknown)`.
23    Unknown,
24}
25
26/// Guaranteed to always typecheck
27pub trait Typeable {
28    fn return_type(&self) -> ReturnType;
29}
30
31impl Display for ReturnType {
32    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
33        match self {
34            ReturnType::Bool => write!(f, "Bool"),
35            ReturnType::Int => write!(f, "Int"),
36            ReturnType::Matrix(inner) => write!(f, "Matrix of {inner}"),
37            ReturnType::Set(inner) => write!(f, "Set of {inner}"),
38            ReturnType::Tuple(types) => {
39                let inners = types.iter().map(|t| format!("{}", t)).join(", ");
40                write!(f, "({inners})")
41            }
42            ReturnType::Record(types) => {
43                let inners = types.iter().map(|t| format!("{}", t)).join(", ");
44                write!(f, "Record {{ {inners} }}")
45            }
46            ReturnType::Function(ty1, ty2) => {
47                write!(f, "Function ({ty1} -> {ty2})")
48            }
49            ReturnType::Unknown => write!(f, "?"),
50        }
51    }
52}
