31/// A comprehension.
32#[derive(Clone, PartialEq, Eq, Uniplate, Serialize, Deserialize, Debug)]
33#[biplate(to=SubModel)]
34#[biplate(to=Expression)]
35#[non_exhaustive]
36pub struct Comprehension {
37    #[doc(hidden)]
38    pub return_expression_submodel: SubModel,
39    #[doc(hidden)]
40    pub generator_submodel: SubModel,
41    #[doc(hidden)]
42    pub induction_vars: Vec<Name>,
43}
44
45impl Comprehension {
46    pub fn domain_of(&self) -> Option<Domain> {
47        let return_expr_domain = self
48            .return_expression_submodel
49            .clone()
50            .into_single_expression()
51            .domain_of()?;
52
53        // return a list (matrix with index domain int(1..)) of return_expr elements
54        Some(Domain::Matrix(
55            Box::new(return_expr_domain),
56            vec![Domain::Int(vec![Range::UnboundedR(1)])],
57        ))
58    }
59
60    pub fn return_expression(self) -> Expression {
61        self.return_expression_submodel.into_single_expression()
62    }
63
64    pub fn replace_return_expression(&mut self, new_expr: Expression) {
65        let new_expr = match new_expr {
66            Expression::And(_, exprs) if (*exprs).clone().unwrap_list().is_some() => {
67                Expression::Root(Metadata::new(), (*exprs).clone().unwrap_list().unwrap())
68            }
69            expr => Expression::Root(Metadata::new(), vec![expr]),
70        };
71
72        *self.return_expression_submodel.root_mut_unchecked() = new_expr;
73    }
74
75    /// Adds a guard to the comprehension. Returns false if the guard does not only reference induction variables.
76    pub fn add_induction_guard(&mut self, guard: Expression) -> bool {
77        if self.is_induction_guard(&guard) {
78            self.generator_submodel.add_constraint(guard);
79            true
80        } else {
81            false
82        }
83    }
84
85    /// True iff expr only references induction variables.
86    pub fn is_induction_guard(&self, expr: &Expression) -> bool {
87        is_induction_guard(&(self.induction_vars.clone().into_iter().collect()), expr)
88    }
89}
90
91impl Display for Comprehension {
92    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
93        let generators: String = self
94            .generator_submodel
95            .symbols()
96            .clone()
97            .into_iter_local()
98            .map(|(name, decl): (Name, DeclarationPtr)| {
99                let domain: Domain = decl.domain().unwrap();
100                (name, domain)
101            })
102            .map(|(name, domain): (Name, Domain)| format!("{name}: {domain}"))
103            .join(",");
104
105        let guards = self
106            .generator_submodel
107            .constraints()
108            .iter()
109            .map(|x| format!("{x}"))
110            .join(",");
111
112        let generators_and_guards = itertools::join([generators, guards], ",");
113
114        let expression = &self.return_expression_submodel;
115        write!(f, "[{expression} | {generators_and_guards}]")
116    }
117}

119/// A builder for a comprehension.
120#[derive(Clone, Debug, PartialEq, Eq)]
121pub struct ComprehensionBuilder {
122    guards: Vec<Expression>,
123    // symbol table containing all the generators
124    // for now, this is just used during parsing - a new symbol table is created using this when we initialise the comprehension
125    // this is not ideal, but i am chucking all this code very soon anyways...
126    generator_symboltable: Rc<RefCell<SymbolTable>>,
127    return_expr_symboltable: Rc<RefCell<SymbolTable>>,
128    induction_variables: BTreeSet<Name>,
129}
130
131impl ComprehensionBuilder {
132    pub fn new(symbol_table_ptr: Rc<RefCell<SymbolTable>>) -> Self {
133        ComprehensionBuilder {
134            guards: vec![],
135            generator_symboltable: Rc::new(RefCell::new(SymbolTable::with_parent(
136                symbol_table_ptr.clone(),
137            ))),
138            return_expr_symboltable: Rc::new(RefCell::new(SymbolTable::with_parent(
139                symbol_table_ptr,
140            ))),
141            induction_variables: BTreeSet::new(),
142        }
143    }
144
145    /// The symbol table for the comprehension generators
146    pub fn generator_symboltable(&mut self) -> Rc<RefCell<SymbolTable>> {
147        Rc::clone(&self.generator_symboltable)
148    }
149
150    /// The symbol table for the comprehension return expression
151    pub fn return_expr_symboltable(&mut self) -> Rc<RefCell<SymbolTable>> {
152        Rc::clone(&self.return_expr_symboltable)
153    }
154
155    pub fn guard(mut self, guard: Expression) -> Self {
156        self.guards.push(guard);
157        self
158    }
159
160    pub fn generator(mut self, declaration: DeclarationPtr) -> Self {
161        let name = declaration.name().clone();
162        let domain = declaration.domain().unwrap();
163        assert!(!self.induction_variables.contains(&name));
164
165        self.induction_variables.insert(name.clone());
166
167        // insert into generator symbol table as a variable
168        (*self.generator_symboltable)
169            .borrow_mut()
170            .insert(declaration);
171
172        // insert into return expression symbol table as a given
173        (*self.return_expr_symboltable)
174            .borrow_mut()
175            .insert(DeclarationPtr::new_given(name, domain));
176
177        self
178    }
179
180    /// Creates a comprehension with the given return expression.
181    ///
182    /// If this comprehension is inside an AC-operator, the kind of this operator should be passed
183    /// in the `comprehension_kind` field.
184    ///
185    /// If a comprehension kind is not given, comprehension guards containing decision variables
186    /// are invalid, and will cause a panic.
187    pub fn with_return_value(
188        self,
189        mut expression: Expression,
190        comprehension_kind: Option<ACOperatorKind>,
191    ) -> Comprehension {
192        let parent_symboltable = self
193            .generator_symboltable
194            .as_ref()
195            .borrow_mut()
196            .parent_mut_unchecked()
197            .clone()
198            .unwrap();
199        let mut generator_submodel = SubModel::new(parent_symboltable.clone());
200        let mut return_expression_submodel = SubModel::new(parent_symboltable);
201
202        *generator_submodel.symbols_ptr_unchecked_mut() = self.generator_symboltable;
203        *return_expression_submodel.symbols_ptr_unchecked_mut() = self.return_expr_symboltable;
204
205        // TODO:also allow guards that reference lettings and givens.
206
207        let induction_variables = self.induction_variables;
208
209        // only guards referencing induction variables can go inside the comprehension
210        let (mut induction_guards, mut other_guards): (Vec<_>, Vec<_>) = self
211            .guards
212            .into_iter()
213            .partition(|x| is_induction_guard(&induction_variables, x));
214
215        let induction_variables_2 = induction_variables.clone();
216        let generator_symboltable_ptr = generator_submodel.symbols_ptr_unchecked().clone();
217
218        // fix induction guard pointers so that they all point to variables in the generator model
219        induction_guards =
220            Biplate::<DeclarationPtr>::transform_bi(&induction_guards, &move |decl| {
221                if induction_variables_2.contains(&decl.name()) {
222                    (*generator_symboltable_ptr)
223                        .borrow()
224                        .lookup_local(&decl.name())
225                        .unwrap()
226                } else {
227                    decl
228                }
229            })
230            .into_iter()
231            .collect_vec();
232
233        let induction_variables_2 = induction_variables.clone();
234        let return_expr_symboltable_ptr =
235            return_expression_submodel.symbols_ptr_unchecked().clone();
236
237        // fix other guard pointers so that they all point to variables in the return expr model
238        other_guards = Biplate::<DeclarationPtr>::transform_bi(&other_guards, &move |decl| {
239            if induction_variables_2.contains(&decl.name()) {
240                (*return_expr_symboltable_ptr)
241                    .borrow()
242                    .lookup_local(&decl.name())
243                    .unwrap()
244            } else {
245                decl
246            }
247        })
248        .into_iter()
249        .collect_vec();
250
251        // handle guards that reference non-induction variables
252        if !other_guards.is_empty() {
253            let comprehension_kind = comprehension_kind.expect(
254                "if any guards reference decision variables, a comprehension kind should be given",
255            );
256
257            let guard_expr = match other_guards.as_slice() {
258                [x] => x.clone(),
259                xs => Expression::And(Metadata::new(), Moo::new(into_matrix_expr!(xs.to_vec()))),
260            };
261
262            expression = match comprehension_kind {
263                ACOperatorKind::And => {
264                    Expression::Imply(Metadata::new(), Moo::new(guard_expr), Moo::new(expression))
265                }
266                ACOperatorKind::Or => Expression::And(
267                    Metadata::new(),
268                    Moo::new(Expression::And(
269                        Metadata::new(),
270                        Moo::new(matrix_expr![guard_expr, expression]),
271                    )),
272                ),
273
274                ACOperatorKind::Sum => {
275                    panic!("guards that reference decision variables not yet implemented for sum");
276                }
277
278                ACOperatorKind::Product => {
279                    panic!(
280                        "guards that reference decision variables not yet implemented for product"
281                    );
282                }
283            }
284        }
285
286        generator_submodel.add_constraints(induction_guards);
287
288        return_expression_submodel.add_constraint(expression);
289
290        Comprehension {
291            return_expression_submodel,
292            generator_submodel,
293            induction_vars: induction_variables.into_iter().collect_vec(),
294        }
295    }
296}

298/// True iff the guard only references induction variables.
299fn is_induction_guard(induction_variables: &BTreeSet<Name>, guard: &Expression) -> bool {
300    guard
301        .universe_bi()
302        .iter()
303        .all(|x| induction_variables.contains(x))
