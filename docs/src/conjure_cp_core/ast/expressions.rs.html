33// Ensure that this type doesn't get too big
34//
35// If you triggered this assertion, you either made a variant of this enum that is too big, or you
36// made Name,Literal,AbstractLiteral,Atom bigger, which made this bigger! To fix this, put some
37// stuff in boxes.
38//
39// Enums take the size of their largest variant, so an enum with mostly small variants and a few
40// large ones wastes memory... A larger Expression type also slows down Oxide.

43// message for 6012de809 (perf: reduce size of AST types, 2025-06-18).

46//

565// for the given matrix literal, return a bounded domain from the min to max of applying op to each
566// child expression.
567//
568// Op must be monotonic.
569//
570// Returns none if unbounded
571fn bounded_i32_domain_for_matrix_literal_monotonic(
572    e: &Expression,
573    op: fn(i32, i32) -> Option<i32>,
574) -> Option<DomainPtr> {
575    // only care about the elements, not the indices
576    let (mut exprs, _) = e.clone().unwrap_matrix_unchecked()?;
577
578    // fold each element's domain into one using op.
579    //
580    // here, I assume that op is monotone. This means that the bounds of op([a1,a2],[b1,b2])  for
581    // the ranges [a1,a2], [b1,b2] will be
582    // [min(op(a1,b1),op(a2,b1),op(a1,b2),op(a2,b2)),max(op(a1,b1),op(a2,b1),op(a1,b2),op(a2,b2))].
583    //
584    // We used to not assume this, and work out the bounds by applying op on the Cartesian product
585    // of A and B; however, this caused a combinatorial explosion and my computer to run out of
586    // memory (on the hakank_eprime_xkcd test)...
587    //Int
588    // For example, to find the bounds of the intervals [1,4], [1,5] combined using op, we used to do
589    //  [min(op(1,1), op(1,2),op(1,3),op(1,4),op(1,5),op(2,1)..
590    //
591    // +,-,/,* are all monotone, so this assumption should be fine for now...
592
593    let expr = exprs.pop()?;
594    let dom = expr.domain_of()?;
595    let Some(GroundDomain::Int(ranges)) = dom.as_ground() else {
596        return None;
597    };
598
599    let (mut current_min, mut current_max) = range_vec_bounds_i32(ranges)?;
600
601    for expr in exprs {
602        let dom = expr.domain_of()?;
603        let Some(GroundDomain::Int(ranges)) = dom.as_ground() else {
604            return None;
605        };
606
607        let (min, max) = range_vec_bounds_i32(ranges)?;
608
609        // all the possible new values for current_min / current_max
610        let minmax = op(min, current_max)?;
611        let minmin = op(min, current_min)?;
612        let maxmin = op(max, current_min)?;
613        let maxmax = op(max, current_max)?;
614        let vals = [minmax, minmin, maxmin, maxmax];
615
616        current_min = *vals
617            .iter()
618            .min()
619            .expect("vals iterator should not be empty, and should have a minimum.");
620        current_max = *vals
621            .iter()
622            .max()
623            .expect("vals iterator should not be empty, and should have a maximum.");
624    }
625
626    if current_min == current_max {
627        Some(Domain::int(vec![Range::Single(current_min)]))
628    } else {
629        Some(Domain::int(vec![Range::Bounded(current_min, current_max)]))
630    }
631}

633// Returns none if unbounded
634fn range_vec_bounds_i32(ranges: &Vec<Range<i32>>) -> Option<(i32, i32)> {
635    let mut min = i32::MAX;
636    let mut max = i32::MIN;
637    for r in ranges {
638        match r {
639            Range::Single(i) => {
640                if *i < min {
641                    min = *i;
642                }
643                if *i > max {
644                    max = *i;
645                }
646            }
647            Range::Bounded(i, j) => {
648                if *i < min {
649                    min = *i;
650                }
651                if *j > max {
652                    max = *j;
653                }
654            }
655            Range::UnboundedR(_) | Range::UnboundedL(_) | Range::Unbounded => return None,
656        }
657    }
658    Some((min, max))
659}

1231impl CategoryOf for Expression {
1232    fn category_of(&self) -> Category {
1233        // take highest category of all the expressions children
1234        let category = self.cata(&move |x,children| {
1235
1236            if let Some(max_category) = children.iter().max() {
1237                // if this expression contains subexpressions, return the maximum category of the
1238                // subexpressions
1239                *max_category
1240            } else {
1241                // this expression has no children
1242                let mut max_category = Category::Bottom;
1243
1244                // calculate the category by looking at all atoms, submodels, comprehensions, and
1245                // declarationptrs inside this expression
1246
1247                // this should generically cover all leaf types we currently have in oxide.
1248
1249                // if x contains submodels (including comprehensions)
1250                if !Biplate::<SubModel>::universe_bi(&x).is_empty() {
1251                    // assume that the category is decision
1252                    return Category::Decision;
1253                }
1254
1255                // if x contains atoms
1256                if let Some(max_atom_category) = Biplate::<Atom>::universe_bi(&x).iter().map(|x| x.category_of()).max()
1257                // and those atoms have a higher category than we already know about
1258                && max_atom_category > max_category{
1259                    // update category 
1260                    max_category = max_atom_category;
1261                }
1262
1263                // if x contains declarationPtrs
1264                if let Some(max_declaration_category) = Biplate::<DeclarationPtr>::universe_bi(&x).iter().map(|x| x.category_of()).max()
1265                // and those pointers have a higher category than we already know about
1266                && max_declaration_category > max_category{
1267                    // update category 
1268                    max_category = max_declaration_category;
1269                }
1270                max_category
1271
1272            }
1273        });
1274
1275        if cfg!(debug_assertions) {
1276            trace!(
1277                category= %category,
1278                expression= %self,
1279                "Called Expression::category_of()"
1280            );
1281        };
1282        category
1283    }
1284}
