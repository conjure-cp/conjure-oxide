32// Ensure that this type doesn't get too big
33//
34// If you triggered this assertion, you either made a variant of this enum that is too big, or you
35// made Name,Literal,AbstractLiteral,Atom bigger, which made this bigger! To fix this, put some
36// stuff in boxes.
37//
38// Enums take the size of their largest variant, so an enum with mostly small variants and a few
39// large ones wastes memory... A larger Expression type also slows down Oxide.
40//
41// For more information, and more details on type sizes and how to measure them, see the commit
42// message for 6012de809 (perf: reduce size of AST types, 2025-06-18).
43//
44// You can also see type sizes in the rustdoc documentation, generated by ./tools/gen_docs.sh
45//
46// https://github.com/conjure-cp/conjure-oxide/commit/6012de8096ca491ded91ecec61352fdf4e994f2e

49// ReturnType, and Metadata contains ReturnType, so Metadata has got bigger. Metadata will get a
50// lot bigger still when we start using it for memoisation, so it should really be
51// boxed ~niklasdewally

494// for the given matrix literal, return a bounded domain from the min to max of applying op to each
495// child expression.
496//
497// Op must be monotonic.
498//
499// Returns none if unbounded
500fn bounded_i32_domain_for_matrix_literal_monotonic(
501    e: &Expression,
502    op: fn(i32, i32) -> Option<i32>,
503) -> Option<Domain> {
504    // only care about the elements, not the indices
505    let (mut exprs, _) = e.clone().unwrap_matrix_unchecked()?;
506
507    // fold each element's domain into one using op.
508    //
509    // here, I assume that op is monotone. This means that the bounds of op([a1,a2],[b1,b2])  for
510    // the ranges [a1,a2], [b1,b2] will be
511    // [min(op(a1,b1),op(a2,b1),op(a1,b2),op(a2,b2)),max(op(a1,b1),op(a2,b1),op(a1,b2),op(a2,b2))].
512    //
513    // We used to not assume this, and work out the bounds by applying op on the Cartesian product
514    // of A and B; however, this caused a combinatorial explosion and my computer to run out of
515    // memory (on the hakank_eprime_xkcd test)...
516    //
517    // For example, to find the bounds of the intervals [1,4], [1,5] combined using op, we used to do
518    //  [min(op(1,1), op(1,2),op(1,3),op(1,4),op(1,5),op(2,1)..
519    //
520    // +,-,/,* are all monotone, so this assumption should be fine for now...
521
522    let expr = exprs.pop()?;
523    let Some(Domain::Int(ranges)) = expr.domain_of() else {
524        return None;
525    };
526
527    let (mut current_min, mut current_max) = range_vec_bounds_i32(&ranges)?;
528
529    for expr in exprs {
530        let Some(Domain::Int(ranges)) = expr.domain_of() else {
531            return None;
532        };
533
534        let (min, max) = range_vec_bounds_i32(&ranges)?;
535
536        // all the possible new values for current_min / current_max
537        let minmax = op(min, current_max)?;
538        let minmin = op(min, current_min)?;
539        let maxmin = op(max, current_min)?;
540        let maxmax = op(max, current_max)?;
541        let vals = [minmax, minmin, maxmin, maxmax];
542
543        current_min = *vals
544            .iter()
545            .min()
546            .expect("vals iterator should not be empty, and should have a minimum.");
547        current_max = *vals
548            .iter()
549            .max()
550            .expect("vals iterator should not be empty, and should have a maximum.");
551    }
552
553    if current_min == current_max {
554        Some(Domain::Int(vec![Range::Single(current_min)]))
555    } else {
556        Some(Domain::Int(vec![Range::Bounded(current_min, current_max)]))
557    }
558}

560// Returns none if unbounded
561fn range_vec_bounds_i32(ranges: &Vec<Range<i32>>) -> Option<(i32, i32)> {
562    let mut min = i32::MAX;
563    let mut max = i32::MIN;
564    for r in ranges {
565        match r {
566            Range::Single(i) => {
567                if *i < min {
568                    min = *i;
569                }
570                if *i > max {
571                    max = *i;
572                }
573            }
574            Range::Bounded(i, j) => {
575                if *i < min {
576                    min = *i;
577                }
578                if *j > max {
579                    max = *j;
580                }
581            }
582            Range::UnboundedR(_) | Range::UnboundedL(_) => return None,
583        }
584    }
585    Some((min, max))
586}

588impl Expression {
589    /// Returns the possible values of the expression, recursing to leaf expressions
590    pub fn domain_of(&self) -> Option<Domain> {
591        let ret = match self {
592            Expression::Union(_, a, b) => Some(Domain::Set(
593                SetAttr::None,
594                Box::new(a.domain_of()?.union(&b.domain_of()?).ok()?),
595            )),
596            Expression::Intersect(_, a, b) => Some(Domain::Set(
597                SetAttr::None,
598                Box::new(a.domain_of()?.intersect(&b.domain_of()?).ok()?),
599            )),
600            Expression::In(_, _, _) => Some(Domain::Bool),
601            Expression::Supset(_, _, _) => Some(Domain::Bool),
602            Expression::SupsetEq(_, _, _) => Some(Domain::Bool),
603            Expression::Subset(_, _, _) => Some(Domain::Bool),
604            Expression::SubsetEq(_, _, _) => Some(Domain::Bool),
605            Expression::AbstractLiteral(_, abslit) => abslit.domain_of(),
606            Expression::DominanceRelation(_, _) => Some(Domain::Bool),
607            Expression::FromSolution(_, expr) => Some(expr.domain_of()),
608            Expression::Metavar(_, _) => None,
609            Expression::Comprehension(_, comprehension) => comprehension.domain_of(),
610            Expression::UnsafeIndex(_, matrix, _) | Expression::SafeIndex(_, matrix, _) => {
611                match matrix.domain_of()? {
612                    Domain::Matrix(elem_domain, _) => Some(*elem_domain),
613                    Domain::Tuple(_) => None,
614                    Domain::Record(_) => None,
615                    _ => {
616                        bug!("subject of an index operation should support indexing")
617                    }
618                }
619            }
620            Expression::UnsafeSlice(_, matrix, indices)
621            | Expression::SafeSlice(_, matrix, indices) => {
622                let sliced_dimension = indices.iter().position(Option::is_none);
623
624                let Domain::Matrix(elem_domain, index_domains) = matrix.domain_of()? else {
625                    bug!("subject of an index operation should be a matrix");
626                };
627
628                match sliced_dimension {
629                    Some(dimension) => Some(Domain::Matrix(
630                        elem_domain,
631                        vec![index_domains[dimension].clone()],
632                    )),
633
634                    // same as index
635                    None => Some(*elem_domain),
636                }
637            }
638            Expression::InDomain(_, _, _) => Some(Domain::Bool),
639            Expression::Atomic(_, Atom::Reference(ptr)) => ptr.domain(),
640            Expression::Atomic(_, atom) => Some(atom.domain_of()),
641            Expression::Scope(_, _) => Some(Domain::Bool),
642            Expression::Sum(_, e) => {
643                bounded_i32_domain_for_matrix_literal_monotonic(e, |x, y| Some(x + y))
644            }
645            Expression::Product(_, e) => {
646                bounded_i32_domain_for_matrix_literal_monotonic(e, |x, y| Some(x * y))
647            }
648            Expression::Min(_, e) => bounded_i32_domain_for_matrix_literal_monotonic(e, |x, y| {
649                Some(if x < y { x } else { y })
650            }),
651            Expression::Max(_, e) => bounded_i32_domain_for_matrix_literal_monotonic(e, |x, y| {
652                Some(if x > y { x } else { y })
653            }),
654            Expression::UnsafeDiv(_, a, b) => a
655                .domain_of()?
656                .apply_i32(
657                    // rust integer division is truncating; however, we want to always round down,
658                    // including for negative numbers.
659                    |x, y| {
660                        if y != 0 {
661                            Some((x as f32 / y as f32).floor() as i32)
662                        } else {
663                            None
664                        }
665                    },
666                    &b.domain_of()?,
667                )
668                .ok(),
669            Expression::SafeDiv(_, a, b) => {
670                // rust integer division is truncating; however, we want to always round down
671                // including for negative numbers.
672                let domain = a.domain_of()?.apply_i32(
673                    |x, y| {
674                        if y != 0 {
675                            Some((x as f32 / y as f32).floor() as i32)
676                        } else {
677                            None
678                        }
679                    },
680                    &b.domain_of()?,
681                );
682
683                match domain {
684                    Ok(Domain::Int(ranges)) => {
685                        let mut ranges = ranges;
686                        ranges.push(Range::Single(0));
687                        Some(Domain::Int(ranges))
688                    }
689                    Err(_) => todo!(),
690                    _ => unreachable!(),
691                }
692            }
693            Expression::UnsafeMod(_, a, b) => a
694                .domain_of()?
695                .apply_i32(
696                    |x, y| if y != 0 { Some(x % y) } else { None },
697                    &b.domain_of()?,
698                )
699                .ok(),
700            Expression::SafeMod(_, a, b) => {
701                let domain = a.domain_of()?.apply_i32(
702                    |x, y| if y != 0 { Some(x % y) } else { None },
703                    &b.domain_of()?,
704                );
705
706                match domain {
707                    Ok(Domain::Int(ranges)) => {
708                        let mut ranges = ranges;
709                        ranges.push(Range::Single(0));
710                        Some(Domain::Int(ranges))
711                    }
712                    Err(_) => todo!(),
713                    _ => unreachable!(),
714                }
715            }
716            Expression::SafePow(_, a, b) | Expression::UnsafePow(_, a, b) => a
717                .domain_of()?
718                .apply_i32(
719                    |x, y| {
720                        if (x != 0 || y != 0) && y >= 0 {
721                            Some(x.pow(y as u32))
722                        } else {
723                            None
724                        }
725                    },
726                    &b.domain_of()?,
727                )
728                .ok(),
729            Expression::Root(_, _) => None,
730            Expression::Bubble(_, inner, _) => inner.domain_of(),
731            Expression::AuxDeclaration(_, _, _) => Some(Domain::Bool),
732            Expression::And(_, _) => Some(Domain::Bool),
733            Expression::Not(_, _) => Some(Domain::Bool),
734            Expression::Or(_, _) => Some(Domain::Bool),
735            Expression::Imply(_, _, _) => Some(Domain::Bool),
736            Expression::Iff(_, _, _) => Some(Domain::Bool),
737            Expression::Eq(_, _, _) => Some(Domain::Bool),
738            Expression::Neq(_, _, _) => Some(Domain::Bool),
739            Expression::Geq(_, _, _) => Some(Domain::Bool),
740            Expression::Leq(_, _, _) => Some(Domain::Bool),
741            Expression::Gt(_, _, _) => Some(Domain::Bool),
742            Expression::Lt(_, _, _) => Some(Domain::Bool),
743            Expression::FlatAbsEq(_, _, _) => Some(Domain::Bool),
744            Expression::FlatSumGeq(_, _, _) => Some(Domain::Bool),
745            Expression::FlatSumLeq(_, _, _) => Some(Domain::Bool),
746            Expression::MinionDivEqUndefZero(_, _, _, _) => Some(Domain::Bool),
747            Expression::MinionModuloEqUndefZero(_, _, _, _) => Some(Domain::Bool),
748            Expression::FlatIneq(_, _, _, _) => Some(Domain::Bool),
749            Expression::AllDiff(_, _) => Some(Domain::Bool),
750            Expression::FlatWatchedLiteral(_, _, _) => Some(Domain::Bool),
751            Expression::MinionReify(_, _, _) => Some(Domain::Bool),
752            Expression::MinionReifyImply(_, _, _) => Some(Domain::Bool),
753            Expression::MinionWInIntervalSet(_, _, _) => Some(Domain::Bool),
754            Expression::MinionWInSet(_, _, _) => Some(Domain::Bool),
755            Expression::MinionElementOne(_, _, _, _) => Some(Domain::Bool),
756            Expression::Neg(_, x) => {
757                let Some(Domain::Int(mut ranges)) = x.domain_of() else {
758                    return None;
759                };
760
761                for range in ranges.iter_mut() {
762                    *range = match range {
763                        Range::Single(x) => Range::Single(-*x),
764                        Range::Bounded(x, y) => Range::Bounded(-*y, -*x),
765                        Range::UnboundedR(i) => Range::UnboundedL(-*i),
766                        Range::UnboundedL(i) => Range::UnboundedR(-*i),
767                    };
768                }
769
770                Some(Domain::Int(ranges))
771            }
772            Expression::Minus(_, a, b) => a
773                .domain_of()?
774                .apply_i32(|x, y| Some(x - y), &b.domain_of()?)
775                .ok(),
776            Expression::FlatAllDiff(_, _) => Some(Domain::Bool),
777            Expression::FlatMinusEq(_, _, _) => Some(Domain::Bool),
778            Expression::FlatProductEq(_, _, _, _) => Some(Domain::Bool),
779            Expression::FlatWeightedSumLeq(_, _, _, _) => Some(Domain::Bool),
780            Expression::FlatWeightedSumGeq(_, _, _, _) => Some(Domain::Bool),
781            Expression::Abs(_, a) => a
782                .domain_of()?
783                .apply_i32(|a, _| Some(a.abs()), &a.domain_of()?)
784                .ok(),
785            Expression::MinionPow(_, _, _, _) => Some(Domain::Bool),
786            Expression::ToInt(_, _) => Some(Domain::Int(vec![Range::Bounded(0, 1)])),
787        };
788        match ret {
789            // TODO: (flm8) the Minion bindings currently only support single ranges for domains, so we use the min/max bounds
790            // Once they support a full domain as we define it, we can remove this conversion
791            Some(Domain::Int(ranges)) if ranges.len() > 1 => {
792                let (min, max) = range_vec_bounds_i32(&ranges)?;
793                Some(Domain::Int(vec![Range::Bounded(min, max)]))
794            }
795            _ => ret,
796        }
797    }

808    /// Checks whether this expression is safe.
809    ///
810    /// An expression is unsafe if can be undefined, or if any of its children can be undefined.
811    ///
812    /// Unsafe expressions are (typically) prefixed with Unsafe in our AST, and can be made
813    /// safe through the use of bubble rules.
814    pub fn is_safe(&self) -> bool {
815        // TODO: memoise in Metadata
816        for expr in self.universe() {
817            match expr {
818                Expression::UnsafeDiv(_, _, _)
819                | Expression::UnsafeMod(_, _, _)
820                | Expression::UnsafePow(_, _, _)
821                | Expression::UnsafeIndex(_, _, _)
822                | Expression::Bubble(_, _, _)
823                | Expression::UnsafeSlice(_, _, _) => {
824                    return false;
825                }
826                _ => {}
827            }
828        }
829        true
830    }

863    /// True iff self and other are both atomic and identical.
864    ///
865    /// This method is useful to cheaply check equivalence. Assuming CSE is enabled, any unifiable
866    /// expressions will be rewritten to a common variable. This is much cheaper than checking the
867    /// entire subtrees of `self` and `other`.
868    pub fn identical_atom_to(&self, other: &Expression) -> bool {
869        let atom1: Result<&Atom, _> = self.try_into();
870        let atom2: Result<&Atom, _> = other.try_into();
871
872        if let (Ok(atom1), Ok(atom2)) = (atom1, atom2) {
873            atom2 == atom1
874        } else {
875            false
876        }
877    }

879    /// If the expression is a list, returns the inner expressions.
880    ///
881    /// A list is any a matrix with the domain `int(1..)`. This includes matrix literals without
882    /// any explicitly specified domain.
883    pub fn unwrap_list(self) -> Option<Vec<Expression>> {
884        match self {
885            Expression::AbstractLiteral(_, matrix @ AbstractLiteral::Matrix(_, _)) => {
886                matrix.unwrap_list().cloned()
887            }
888            Expression::Atomic(
889                _,
890                Atom::Literal(Literal::AbstractLiteral(matrix @ AbstractLiteral::Matrix(_, _))),
891            ) => matrix.unwrap_list().map(|elems| {
892                elems
893                    .clone()
894                    .into_iter()
895                    .map(|x: Literal| Expression::Atomic(Metadata::new(), Atom::Literal(x)))
896                    .collect_vec()
897            }),
898            _ => None,
899        }
900    }

902    /// If the expression is a matrix, gets it elements and index domain.
903    ///
904    /// **Consider using the safer [`Expression::unwrap_list`] instead.**
905    ///
906    /// It is generally undefined to edit the length of a matrix unless it is a list (as defined by
907    /// [`Expression::unwrap_list`]). Users of this function should ensure that, if the matrix is
908    /// reconstructed, the index domain and the number of elements in the matrix remain the same.
909    pub fn unwrap_matrix_unchecked(self) -> Option<(Vec<Expression>, Domain)> {
910        match self {
911            Expression::AbstractLiteral(_, AbstractLiteral::Matrix(elems, domain)) => {
912                Some((elems, *domain))
913            }
914            Expression::Atomic(
915                _,
916                Atom::Literal(Literal::AbstractLiteral(AbstractLiteral::Matrix(elems, domain))),
917            ) => Some((
918                elems
919                    .into_iter()
920                    .map(|x: Literal| Expression::Atomic(Metadata::new(), Atom::Literal(x)))
921                    .collect_vec(),
922                *domain,
923            )),
924
925            _ => None,
926        }
927    }

1027impl CategoryOf for Expression {
1028    fn category_of(&self) -> Category {
1029        // take highest category of all the expressions children
1030        let category = self.cata(&move |x,children| {
1031
1032            if let Some(max_category) = children.iter().max() {
1033                // if this expression contains subexpressions, return the maximum category of the
1034                // subexpressions
1035                *max_category
1036            } else {
1037                // this expression has no children
1038                let mut max_category = Category::Bottom;
1039
1040                // calculate the category by looking at all atoms, submodels, comprehensions, and
1041                // declarationptrs inside this expression
1042
1043                // this should generically cover all leaf types we currently have in oxide.
1044
1045                // if x contains submodels (including comprehensions)
1046                if !Biplate::<SubModel>::universe_bi(&x).is_empty() {
1047                    // assume that the category is decision
1048                    return Category::Decision;
1049                }
1050
1051                // if x contains atoms
1052                if let Some(max_atom_category) = Biplate::<Atom>::universe_bi(&x).iter().map(|x| x.category_of()).max()
1053                // and those atoms have a higher category than we already know about
1054                && max_atom_category > max_category{
1055                    // update category 
1056                    max_category = max_atom_category;
1057                }
1058
1059                // if x contains declarationPtrs
1060                if let Some(max_declaration_category) = Biplate::<DeclarationPtr>::universe_bi(&x).iter().map(|x| x.category_of()).max()
1061                // and those pointers have a higher category than we already know about
1062                && max_declaration_category > max_category{
1063                    // update category 
1064                    max_category = max_declaration_category;
1065                }
1066                max_category
1067
1068            }
1069        });
1070
1071        if cfg!(debug_assertions) {
1072            trace!(
1073                category= %category,
1074                expression= %self,
1075                "Called Expression::category_of()"
1076            );
1077        };
1078        category
1079    }
1080}

1312impl Typeable for Expression {
1313    fn return_type(&self) -> Option<ReturnType> {
1314        match self {
1315            Expression::Union(_, subject, _) => {
1316                Some(ReturnType::Set(Box::new(subject.return_type()?)))
1317            }
1318            Expression::Intersect(_, subject, _) => {
1319                Some(ReturnType::Set(Box::new(subject.return_type()?)))
1320            }
1321            Expression::In(_, _, _) => Some(ReturnType::Bool),
1322            Expression::Supset(_, _, _) => Some(ReturnType::Bool),
1323            Expression::SupsetEq(_, _, _) => Some(ReturnType::Bool),
1324            Expression::Subset(_, _, _) => Some(ReturnType::Bool),
1325            Expression::SubsetEq(_, _, _) => Some(ReturnType::Bool),
1326            Expression::AbstractLiteral(_, lit) => lit.return_type(),
1327            Expression::UnsafeIndex(_, subject, _) | Expression::SafeIndex(_, subject, _) => {
1328                let mut elem_typ = subject.return_type()?;
1329                let ReturnType::Matrix(_) = elem_typ else {
1330                    return None;
1331                };
1332
1333                // unwrap the return types of n-d matrices to get to the real element typetype.
1334                while let ReturnType::Matrix(new_elem_typ) = elem_typ {
1335                    elem_typ = *new_elem_typ;
1336                }
1337
1338                Some(elem_typ)
1339            }
1340            Expression::UnsafeSlice(_, subject, _) | Expression::SafeSlice(_, subject, _) => {
1341                Some(ReturnType::Matrix(Box::new(subject.return_type()?)))
1342            }
1343            Expression::InDomain(_, _, _) => Some(ReturnType::Bool),
1344            Expression::Comprehension(_, _) => None,
1345            Expression::Root(_, _) => Some(ReturnType::Bool),
1346            Expression::DominanceRelation(_, _) => Some(ReturnType::Bool),
1347            Expression::FromSolution(_, expr) => expr.return_type(),
1348            Expression::Metavar(_, _) => None,
1349            Expression::Atomic(_, atom) => atom.return_type(),
1350            Expression::Scope(_, scope) => scope.return_type(),
1351            Expression::Abs(_, _) => Some(ReturnType::Int),
1352            Expression::Sum(_, _) => Some(ReturnType::Int),
1353            Expression::Product(_, _) => Some(ReturnType::Int),
1354            Expression::Min(_, _) => Some(ReturnType::Int),
1355            Expression::Max(_, _) => Some(ReturnType::Int),
1356            Expression::Not(_, _) => Some(ReturnType::Bool),
1357            Expression::Or(_, _) => Some(ReturnType::Bool),
1358            Expression::Imply(_, _, _) => Some(ReturnType::Bool),
1359            Expression::Iff(_, _, _) => Some(ReturnType::Bool),
1360            Expression::And(_, _) => Some(ReturnType::Bool),
1361            Expression::Eq(_, _, _) => Some(ReturnType::Bool),
1362            Expression::Neq(_, _, _) => Some(ReturnType::Bool),
1363            Expression::Geq(_, _, _) => Some(ReturnType::Bool),
1364            Expression::Leq(_, _, _) => Some(ReturnType::Bool),
1365            Expression::Gt(_, _, _) => Some(ReturnType::Bool),
1366            Expression::Lt(_, _, _) => Some(ReturnType::Bool),
1367            Expression::SafeDiv(_, _, _) => Some(ReturnType::Int),
1368            Expression::UnsafeDiv(_, _, _) => Some(ReturnType::Int),
1369            Expression::FlatAllDiff(_, _) => Some(ReturnType::Bool),
1370            Expression::FlatSumGeq(_, _, _) => Some(ReturnType::Bool),
1371            Expression::FlatSumLeq(_, _, _) => Some(ReturnType::Bool),
1372            Expression::MinionDivEqUndefZero(_, _, _, _) => Some(ReturnType::Bool),
1373            Expression::FlatIneq(_, _, _, _) => Some(ReturnType::Bool),
1374            Expression::AllDiff(_, _) => Some(ReturnType::Bool),
1375            Expression::Bubble(_, inner, _) => inner.return_type(),
1376            Expression::FlatWatchedLiteral(_, _, _) => Some(ReturnType::Bool),
1377            Expression::MinionReify(_, _, _) => Some(ReturnType::Bool),
1378            Expression::MinionReifyImply(_, _, _) => Some(ReturnType::Bool),
1379            Expression::MinionWInIntervalSet(_, _, _) => Some(ReturnType::Bool),
1380            Expression::MinionWInSet(_, _, _) => Some(ReturnType::Bool),
1381            Expression::MinionElementOne(_, _, _, _) => Some(ReturnType::Bool),
1382            Expression::AuxDeclaration(_, _, _) => Some(ReturnType::Bool),
1383            Expression::UnsafeMod(_, _, _) => Some(ReturnType::Int),
1384            Expression::SafeMod(_, _, _) => Some(ReturnType::Int),
1385            Expression::MinionModuloEqUndefZero(_, _, _, _) => Some(ReturnType::Bool),
1386            Expression::Neg(_, _) => Some(ReturnType::Int),
1387            Expression::UnsafePow(_, _, _) => Some(ReturnType::Int),
1388            Expression::SafePow(_, _, _) => Some(ReturnType::Int),
1389            Expression::Minus(_, _, _) => Some(ReturnType::Int),
1390            Expression::FlatAbsEq(_, _, _) => Some(ReturnType::Bool),
1391            Expression::FlatMinusEq(_, _, _) => Some(ReturnType::Bool),
1392            Expression::FlatProductEq(_, _, _, _) => Some(ReturnType::Bool),
1393            Expression::FlatWeightedSumLeq(_, _, _, _) => Some(ReturnType::Bool),
1394            Expression::FlatWeightedSumGeq(_, _, _, _) => Some(ReturnType::Bool),
1395            Expression::MinionPow(_, _, _, _) => Some(ReturnType::Bool),
1396            Expression::ToInt(_, _) => Some(ReturnType::Int),
1397        }
1398    }
1399}
