25pub fn main() {
26    // exit with 2 instead of 1 on failure,like grep
27    match run() {
28        Ok(_) => {
29            exit(0);
30        }
31        Err(e) => {
32            eprintln!("{e:?}");
33            exit(2);
34        }
35    }
36}

38pub fn run() -> anyhow::Result<()> {
39    let cli = Cli::parse();
40
41    if cli.version {
42        println!("Version: {}", git_version!());
43        return Ok(());
44    }
45
46    setup_logging(&cli.global_args)?;
47
48    run_subcommand(cli)
49}

51fn setup_logging(global_args: &GlobalArgs) -> anyhow::Result<()> {
52    // Logging:
53    //
54    // Using `tracing` framework, but this automatically reads stuff from `log`.
55    //
56    // A Subscriber is responsible for logging.
57    //
58    // It consists of composable layers, each of which logs to a different place in a different
59    // format.
60    let json_log_file = File::options()
61        .create(true)
62        .append(true)
63        .open("conjure_oxide_log.json")?;
64
65    let log_file = File::options()
66        .create(true)
67        .append(true)
68        .open("conjure_oxide.log")?;
69
70    // get log level from env-var RUST_LOG
71
72    let json_layer = tracing_subscriber::fmt::layer()
73        .json()
74        .with_writer(Arc::new(json_log_file))
75        .with_filter(LevelFilter::TRACE);
76
77    let file_layer = tracing_subscriber::fmt::layer()
78        .compact()
79        .with_ansi(false)
80        .with_writer(Arc::new(log_file))
81        .with_filter(LevelFilter::TRACE);
82
83    let default_stderr_level = if global_args.verbose {
84        LevelFilter::DEBUG
85    } else {
86        LevelFilter::WARN
87    };
88
89    let env_filter = EnvFilter::builder()
90        .with_default_directive(default_stderr_level.into())
91        .from_env_lossy();
92
93    let stderr_layer = if global_args.verbose {
94        Layer::boxed(
95            tracing_subscriber::fmt::layer()
96                .pretty()
97                .with_writer(Arc::new(std::io::stderr()))
98                .with_ansi(true)
99                .with_filter(env_filter),
100        )
101    } else {
102        Layer::boxed(
103            tracing_subscriber::fmt::layer()
104                .compact()
105                .with_writer(Arc::new(std::io::stderr()))
106                .with_ansi(true)
107                .with_filter(env_filter),
108        )
109    };
110
111    let human_rule_trace_layer = global_args.human_rule_trace.clone().map(|x| {
112        let file = File::create(x).expect("Unable to create rule trace file");
113        fmt::layer()
114            .with_writer(file)
115            .with_level(false)
116            .without_time()
117            .with_target(false)
118            .with_filter(EnvFilter::new("rule_engine_human=trace"))
119            .with_filter(FilterFn::new(|meta| meta.target() == "rule_engine_human"))
120    });
121    // load the loggers
122    tracing_subscriber::registry()
123        .with(json_layer)
124        .with(stderr_layer)
125        .with(file_layer)
126        .with(human_rule_trace_layer)
127        .init();
128
129    Ok(())
130}

132fn run_completion_command(completion_args: cli::CompletionArgs) -> anyhow::Result<()> {
133    let mut cmd = Cli::command();
134    let shell = completion_args.shell;
135    let name = cmd.get_name().to_string();
136
137    eprintln!("Generating completion for {shell}...");
138
139    generate(shell, &mut cmd, name, &mut io::stdout());
140    Ok(())
141}
142
143/// Runs the selected subcommand
144fn run_subcommand(cli: Cli) -> anyhow::Result<()> {
145    let global_args = cli.global_args;
146    match cli.subcommand {
147        cli::Command::Solve(solve_args) => run_solve_command(global_args, solve_args),
148        cli::Command::TestSolve(local_args) => run_test_solve_command(global_args, local_args),
149        cli::Command::PrintJsonSchema => run_print_info_schema_command(),
150        cli::Command::Completion(completion_args) => run_completion_command(completion_args),
151    }
152}
