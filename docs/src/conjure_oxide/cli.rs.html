15/// All subcommands of conjure-oxide
16#[derive(Clone, Debug, Subcommand)]
17pub enum Command {
18    /// Solve a model
19    Solve(solve::Args),
20    /// Print the JSON info file schema
21    PrintJsonSchema,
22    /// Tests whether the Essence model is solvable with Conjure Oxide, and whether it gets the
23    /// same solutions as Conjure.
24    ///
25    /// Return-code will be 0 if the solutions match, 1 if they don't, and >1 on crash.
26    TestSolve(test_solve::Args),
27    /// Generate a completion script for the shell provided
28    Completion(CompletionArgs),
29    Pretty(pretty::Args),
30    // Run the language server
31    ServerLSP,
32}

53#[derive(Debug, Clone, Args)]
54pub struct GlobalArgs {
55    /// Extra rule sets to enable
56    #[arg(long, value_name = "EXTRA_RULE_SETS", global = true)]
57    pub extra_rule_sets: Vec<String>,
58
59    /// Solver family to use
60    #[arg(
61        long,
62        value_name = "SOLVER",
63        default_value_t = SolverFamily::Minion,
64        short = 's',
65        global = true
66    )]
67    pub solver: SolverFamily,
68
69    /// Log verbosely
70    #[arg(long, short = 'v', help = "Log verbosely to sterr", global = true, help_heading = LOGGING_HELP_HEADING)]
71    pub verbose: bool,
72
73    // --no-x flag disables --x flag : https://jwodder.github.io/kbits/posts/clap-bool-negate/
74    /// Check for multiple equally applicable rules, exiting if any are found.
75    ///
76    /// Only compatible with the default rewriter.
77    #[arg(
78        long,
79        overrides_with = "_no_check_equally_applicable_rules",
80        default_value_t = false,
81        global = true,
82        help_heading= DEBUG_HELP_HEADING
83    )]
84    pub check_equally_applicable_rules: bool,
85
86    /// Output file for the human readable rule trace.
87    #[arg(long, global = true, help_heading=LOGGING_HELP_HEADING)]
88    pub human_rule_trace: Option<PathBuf>,
89
90    /// Do not check for multiple equally applicable rules [default].
91    ///
92    /// Only compatible with the default rewriter.
93    #[arg(long, global = true, help_heading = DEBUG_HELP_HEADING)]
94    pub _no_check_equally_applicable_rules: bool,
95
96    /// Use the native parser instead of Conjure's.
97    #[arg(long, default_value_t = false, global = true, help_heading = EXPERIMENTAL_HELP_HEADING)]
98    pub use_native_parser: bool,
99
100    /// Do not use better comprehension expanding for ac operators (and,or,sum,product).
101    #[arg(long, default_value_t = false, global=true, help_heading = OPTIMISATIONS_HELP_HEADING)]
102    pub no_use_expand_ac: bool,
103
104    /// Save a solver input file to <filename>.
105    ///
106    /// This input file will be in a format compatible by the command-line
107    /// interface of the selected solver. For example, when the solver is Minion,
108    /// a valid .minion file will be output.
109    ///
110    /// This file is for informational purposes only; the results of running
111    /// this file cannot be used by Conjure Oxide in any way.
112    #[arg(long,global=true, value_names=["filename"], next_line_help=true, help_heading=LOGGING_HELP_HEADING)]
113    pub save_solver_input_file: Option<PathBuf>,
114
115    /// Use the experimental optimized / dirty-clean rewriter, instead of the default rewriter
116    #[arg(long, default_value_t = false, global = true, help_heading = EXPERIMENTAL_HELP_HEADING)]
117    pub use_optimised_rewriter: bool,
118
119    /// Exit after all comprehensions have been unrolled, printing the number of expressions at that point.
120    ///
121    /// This is only compatible with the default rewriter.
122    ///
123    /// This flag is useful to compare how comprehension optimisations, such as expand-ac, effect
124    /// rewriting.
125    #[arg(long, default_value_t = false, global = true, help_heading = DEBUG_HELP_HEADING)]
126    pub exit_after_unrolling: bool,
127
128    /// Stop the solver after the given timeout.
129    ///
130    /// Currently only SMT supports this feature.
131    #[arg(long, global = true, help_heading = OPTIMISATIONS_HELP_HEADING)]
132    pub solver_timeout: Option<humantime::Duration>,
133}
