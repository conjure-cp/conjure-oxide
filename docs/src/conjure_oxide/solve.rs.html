58pub fn run_solve_command(global_args: GlobalArgs, solve_args: Args) -> anyhow::Result<()> {
59    let input_file = solve_args.input_file.clone();
60
61    // each step is in its own method so that similar commands
62    // (e.g. testsolve) can reuse some of these steps.
63
64    let context = init_context(&global_args, input_file)?;
65    let model = parse(&global_args, Arc::clone(&context))?;
66    let rewritten_model = rewrite(model, &global_args, Arc::clone(&context))?;
67    let solver = init_solver(&global_args);
68
69    if solve_args.no_run_solver {
70        println!("{}", &rewritten_model);
71
72        if let Some(path) = global_args.save_solver_input_file {
73            let solver = solver.load_model(rewritten_model)?;
74            eprintln!("Writing solver input file to {}", path.display());
75            let mut file: Box<dyn std::io::Write> = Box::new(File::create(path)?);
76            solver.write_solver_input_file(&mut file)?;
77        }
78    } else {
79        run_solver(solver, &global_args, &solve_args, rewritten_model)?
80    }
81
82    // still do postamble even if we didn't run the solver
83    if let Some(ref path) = solve_args.info_json_path {
84        let context_obj = context.read().unwrap().clone();
85        let generated_json = &serde_json::to_value(context_obj)?;
86        let pretty_json = serde_json::to_string_pretty(&generated_json)?;
87        File::create(path)?.write_all(pretty_json.as_bytes())?;
88    }
89    Ok(())
90}

92/// Returns a new Context and Solver for solving.
93pub(crate) fn init_context(
94    global_args: &GlobalArgs,
95    input_file: PathBuf,
96) -> anyhow::Result<Arc<RwLock<Context<'static>>>> {
97    let target_family = global_args.solver;
98    let mut extra_rule_sets: Vec<&str> = DEFAULT_RULE_SETS.to_vec();
99    for rs in &global_args.extra_rule_sets {
100        extra_rule_sets.push(rs.as_str());
101    }
102
103    if global_args.no_use_expand_ac {
104        extra_rule_sets.pop_if(|x| x == &"Better_AC_Comprehension_Expansion");
105    }
106
107    let rule_sets = match resolve_rule_sets(target_family, &extra_rule_sets) {
108        Ok(rs) => rs,
109        Err(e) => {
110            tracing::error!("Error resolving rule sets: {}", e);
111            exit(1);
112        }
113    };
114
115    let pretty_rule_sets = rule_sets
116        .iter()
117        .map(|rule_set| rule_set.name)
118        .collect::<Vec<_>>()
119        .join(", ");
120
121    tracing::info!("Enabled rule sets: [{}]", pretty_rule_sets);
122    tracing::info!(
123        target: "file",
124        "Rule sets: {}",
125        pretty_rule_sets
126    );
127
128    let rules = get_rules(&rule_sets)?.into_iter().collect::<Vec<_>>();
129    tracing::info!(
130        target: "file",
131        "Rules: {}",
132        rules.iter().map(|rd| format!("{rd}")).collect::<Vec<_>>().join("\n")
133    );
134    let context = Context::new_ptr(
135        target_family,
136        extra_rule_sets.iter().map(|rs| rs.to_string()).collect(),
137        rules,
138        rule_sets.clone(),
139    );
140
141    context.write().unwrap().file_name = Some(input_file.to_str().expect("").into());
142
143    Ok(context)
144}

146pub(crate) fn init_solver(global_args: &GlobalArgs) -> Solver {
147    let family = global_args.solver;
148    let solver_timeout = global_args
149        .solver_timeout
150        .map(|dur| Duration::from(dur).as_millis());
151
152    match family {
153        SolverFamily::Minion => Solver::new(Minion::default()),
154        SolverFamily::Sat => Solver::new(Sat::default()),
155        #[cfg(feature = "smt")]
156        SolverFamily::Smt(theory_cfg) => {
157            let timeout_ms = solver_timeout.map(|ms| u64::try_from(ms).expect("Timeout too large"));
158            Solver::new(Smt::new(timeout_ms, theory_cfg))
159        }
160    }
161}

163pub(crate) fn parse(
164    global_args: &GlobalArgs,
165    context: Arc<RwLock<Context<'static>>>,
166) -> anyhow::Result<Model> {
167    let input_file: String = context
168        .read()
169        .unwrap()
170        .file_name
171        .clone()
172        .expect("context should contain the input file");
173
174    tracing::info!(target: "file", "Input file: {}", input_file);
175    if global_args.use_native_parser {
176        parse_essence_file_native(input_file.as_str(), context.clone()).map_err(|e| e.into())
177    } else {
178        conjure_executable()
179            .map_err(|e| anyhow!("Could not find correct conjure executable: {e}"))?;
180
181        let mut cmd = std::process::Command::new("conjure");
182        let output = cmd
183            .arg("pretty")
184            .arg("--output-format=astjson")
185            .arg(input_file)
186            .output()?;
187
188        let conjure_stderr = String::from_utf8(output.stderr)?;
189
190        ensure!(conjure_stderr.is_empty(), conjure_stderr);
191
192        let astjson = String::from_utf8(output.stdout)?;
193
194        if cfg!(feature = "extra-rule-checks") {
195            tracing::info!("extra-rule-checks: enabled");
196        } else {
197            tracing::info!("extra-rule-checks: disabled");
198        }
199
200        model_from_json(&astjson, context.clone()).map_err(|e| anyhow!(e))
201    }
202}

204pub(crate) fn rewrite(
205    model: Model,
206    global_args: &GlobalArgs,
207    context: Arc<RwLock<Context<'static>>>,
208) -> anyhow::Result<Model> {
209    tracing::info!("Initial model: \n{}\n", model);
210
211    let rule_sets = context.read().unwrap().rule_sets.clone();
212
213    let new_model = if global_args.use_optimised_rewriter {
214        USE_OPTIMISED_REWRITER_FOR_COMPREHENSIONS.store(true, std::sync::atomic::Ordering::Relaxed);
215        tracing::info!("Rewriting the model using the optimising rewriter");
216        rewrite_morph(
217            model,
218            &rule_sets,
219            global_args.check_equally_applicable_rules,
220        )
221    } else {
222        tracing::info!("Rewriting the model using the default / naive rewriter");
223        if global_args.exit_after_unrolling {
224            tracing::info!("Exiting after unrolling");
225        }
226        rewrite_naive(
227            &model,
228            &rule_sets,
229            global_args.check_equally_applicable_rules,
230            global_args.exit_after_unrolling,
231        )?
232    };
233
234    tracing::info!("Rewritten model: \n{}\n", new_model);
235    Ok(new_model)
236}

238fn run_solver(
239    solver: Solver,
240    global_args: &GlobalArgs,
241    cmd_args: &Args,
242    model: Model,
243) -> anyhow::Result<()> {
244    let out_file: Option<File> = match &cmd_args.output {
245        None => None,
246        Some(pth) => Some(
247            File::options()
248                .create(true)
249                .truncate(true)
250                .write(true)
251                .open(pth)?,
252        ),
253    };
254
255    let solutions = get_solutions(
256        solver,
257        model,
258        cmd_args.number_of_solutions,
259        &global_args.save_solver_input_file,
260    )?;
261    tracing::info!(target: "file", "Solutions: {}", solutions_to_json(&solutions));
262
263    let solutions_json = solutions_to_json(&solutions);
264    let solutions_str = to_string_pretty(&solutions_json)?;
265    match out_file {
266        None => {
267            println!("Solutions:");
268            println!("{solutions_str}");
269        }
270        Some(mut outf) => {
271            outf.write_all(solutions_str.as_bytes())?;
272            println!(
273                "Solutions saved to {:?}",
274                &cmd_args.output.clone().unwrap().canonicalize()?
275            )
276        }
277    }
278    Ok(())
