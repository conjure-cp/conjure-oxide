16/// Parse an Essence file into a Model using the tree-sitter parser.
17pub fn parse_essence_file_native(
18    path: &str,
19    context: Arc<RwLock<Context<'static>>>,
20) -> Result<Model, EssenceParseError> {
21    let source_code = fs::read_to_string(path)
22        .unwrap_or_else(|_| panic!("Failed to read the source code file {path}"));
23    parse_essence_with_context(&source_code, context)
24}

26pub fn parse_essence_with_context(
27    src: &str,
28    context: Arc<RwLock<Context<'static>>>,
29) -> Result<Model, EssenceParseError> {
30    let (tree, source_code) = match get_tree(src) {
31        Some(tree) => tree,
32        None => {
33            return Err(EssenceParseError::TreeSitterError(
34                "Failed to parse source code".to_string(),
35            ));
36        }
37    };
38
39    let mut model = Model::new(context);
40    // let symbols = model.as_submodel().symbols().clone();
41    let root_node = tree.root_node();
42    for statement in named_children(&root_node) {
43        match statement.kind() {
44            "single_line_comment" => {}
45            "language_declaration" => {}
46            "find_statement" => {
47                let var_hashmap = parse_find_statement(statement, &source_code)?;
48                for (name, domain) in var_hashmap {
49                    model
50                        .as_submodel_mut()
51                        .symbols_mut()
52                        .insert(DeclarationPtr::new_var(name, domain));
53                }
54            }
55            "bool_expr" | "atom" | "comparison_expr" => {
56                let current_symbols = model.as_submodel().symbols().clone();
57
58                model.as_submodel_mut().add_constraint(parse_expression(
59                    statement,
60                    &source_code,
61                    &statement,
62                    Some(&current_symbols),
63                )?);
64            }
65            "language_label" => {}
66            "letting_statement" => {
67                let current_symbols = model.as_submodel().symbols().clone();
68                let letting_vars =
69                    parse_letting_statement(statement, &source_code, Some(&current_symbols))?;
70                model.as_submodel_mut().symbols_mut().extend(letting_vars);
71            }
72            "dominance_relation" => {
73                let inner = statement
74                    .child_by_field_name("expression")
75                    .expect("Expected a sub-expression inside `dominanceRelation`");
76                let current_symbols = model.as_submodel().symbols().clone();
77                let expr =
78                    parse_expression(inner, &source_code, &statement, Some(&current_symbols))?;
79                let dominance = Expression::DominanceRelation(Metadata::new(), Moo::new(expr));
80                if model.dominance.is_some() {
81                    return Err(EssenceParseError::syntax_error(
82                        "Duplicate dominance relation".to_string(),
83                        None,
84                    ));
85                }
86                model.dominance = Some(dominance);
87            }
88            "ERROR" => {
89                let raw_expr = &source_code[statement.start_byte()..statement.end_byte()];
90                return Err(EssenceParseError::syntax_error(
91                    format!("'{raw_expr}' is not a valid expression"),
92                    Some(statement.range()),
93                ));
94            }
95            _ => {
96                let kind = statement.kind();
97                return Err(EssenceParseError::syntax_error(
98                    format!("Unrecognized top level statement kind: {kind}"),
99                    Some(statement.range()),
100                ));
101            }
102        }
103    }
104    Ok(model)
105}

107pub fn parse_essence(src: &str) -> Result<Model, EssenceParseError> {
108    let context = Arc::new(RwLock::new(Context::default()));
109    parse_essence_with_context(src, context)
110}

122    #[test]
123    pub fn test_parse_xyz() {
124        let src = "
125        find x, y, z : int(1..4)
126        such that x + y + z = 4
127        such that x >= y
128        ";
129
130        let model = parse_essence(src).unwrap();
131
132        let st = model.as_submodel().symbols();
133        let x = st.lookup(&Name::user("x")).unwrap();
134        let y = st.lookup(&Name::user("y")).unwrap();
135        let z = st.lookup(&Name::user("z")).unwrap();
136        assert_eq!(x.domain(), Some(domain_int!(1..4)));
137        assert_eq!(y.domain(), Some(domain_int!(1..4)));
138        assert_eq!(z.domain(), Some(domain_int!(1..4)));
139
140        let constraints = model.as_submodel().constraints();
141        assert_eq!(constraints.len(), 2);
142
143        let c1 = constraints[0].clone();
144        let x_e = Expression::Atomic(Metadata::new(), Atom::new_ref(x));
145        let y_e = Expression::Atomic(Metadata::new(), Atom::new_ref(y));
146        let z_e = Expression::Atomic(Metadata::new(), Atom::new_ref(z));
147        assert_eq!(
148            c1,
149            Expression::Eq(
150                Metadata::new(),
151                Moo::new(Expression::Sum(
152                    Metadata::new(),
153                    Moo::new(matrix_expr!(
154                        Expression::Sum(
155                            Metadata::new(),
156                            Moo::new(matrix_expr!(x_e.clone(), y_e.clone()))
157                        ),
158                        z_e
159                    ))
160                )),
161                Moo::new(Expression::Atomic(Metadata::new(), 4.into()))
162            )
163        );
164
165        let c2 = constraints[1].clone();
166        assert_eq!(
167            c2,
168            Expression::Geq(Metadata::new(), Moo::new(x_e), Moo::new(y_e))
169        );
170    }

172    #[test]
173    pub fn test_parse_letting_index() {
174        let src = "
175        letting a be [ [ 1,2,3 ; int(1,2,4) ], [ 1,3,2 ; int(1,2,4) ], [ 3,2,1 ; int(1,2,4) ] ; int(-2..0) ]
176        find b: int(1..5)
177        such that
178        b < a[-2,2],
179        allDiff(a[-2,..])
180        ";
181
182        let model = parse_essence(src).unwrap();
183        let st = model.as_submodel().symbols();
184        let a_decl = st.lookup(&Name::user("a")).unwrap();
185        let a = a_decl.as_value_letting().unwrap().deref().clone();
186        assert_eq!(
187            a,
188            matrix_expr!(
189                matrix_expr!(1.into(), 2.into(), 3.into() ; domain_int!(1, 2, 4)),
190                matrix_expr!(1.into(), 3.into(), 2.into() ; domain_int!(1, 2, 4)),
191                matrix_expr!(3.into(), 2.into(), 1.into() ; domain_int!(1, 2, 4));
192                domain_int!(-2..0)
193            )
194        )
195    }
