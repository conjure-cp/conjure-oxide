9pub fn parse_atom(
10    node: &Node,
11    source_code: &str,
12    root: &Node,
13    symbols: Option<&SymbolTable>,
14) -> Result<Expression, EssenceParseError> {
15    match node.kind() {
16        "atom" => parse_atom(&named_child!(node), source_code, root, symbols),
17        "metavar" => {
18            let ident = field!(node, "identifier");
19            let name_str = &source_code[ident.start_byte()..ident.end_byte()];
20            Ok(Expression::Metavar(Metadata::new(), Ustr::from(name_str)))
21        }
22        "identifier" => parse_variable(node, source_code, symbols)
23            .map(|var| Expression::Atomic(Metadata::new(), var)),
24        "from_solution" => {
25            if root.kind() != "dominance_relation" {
26                return Err(EssenceParseError::syntax_error(
27                    "fromSolution only allowed inside dominance relations".to_string(),
28                    Some(node.range()),
29                ));
30            }
31
32            let inner = parse_variable(&field!(node, "variable"), source_code, symbols)?;
33            Ok(Expression::FromSolution(Metadata::new(), Moo::new(inner)))
34        }
35        "constant" => {
36            let lit = parse_constant(node, source_code)?;
37            Ok(Expression::Atomic(Metadata::new(), Atom::Literal(lit)))
38        }
39        "matrix" | "record" | "tuple" => parse_abstract(node, source_code, symbols)
40            .map(|l| Expression::AbstractLiteral(Metadata::new(), l)),
41        "tuple_matrix_record_index_or_slice" => {
42            parse_index_or_slice(node, source_code, root, symbols)
43        }
44        _ => Err(EssenceParseError::syntax_error(
45            format!("Expected atom, got: {}", node.kind()),
46            Some(node.range()),
47        )),
48    }
49}

51fn parse_index_or_slice(
52    node: &Node,
53    source_code: &str,
54    root: &Node,
55    symbols: Option<&SymbolTable>,
56) -> Result<Expression, EssenceParseError> {
57    let collection = parse_atom(&field!(node, "tuple_or_matrix"), source_code, root, symbols)?;
58    let mut indices = Vec::new();
59    for idx_node in named_children(&field!(node, "indices")) {
60        indices.push(parse_index(&idx_node, source_code, symbols)?);
61    }
62
63    let has_null_idx = indices.iter().any(|idx| idx.is_none());
64    // TODO: We could check whether the slice/index is safe here
65    if has_null_idx {
66        // It's a slice
67        Ok(Expression::UnsafeSlice(
68            Metadata::new(),
69            Moo::new(collection),
70            indices,
71        ))
72    } else {
73        // It's an index
74        let idx_exprs: Vec<Expression> = indices.into_iter().map(|idx| idx.unwrap()).collect();
75        Ok(Expression::UnsafeIndex(
76            Metadata::new(),
77            Moo::new(collection),
78            idx_exprs,
79        ))
80    }
81}

83fn parse_index(
84    node: &Node,
85    source_code: &str,
86    symbols: Option<&SymbolTable>,
87) -> Result<Option<Expression>, EssenceParseError> {
88    match node.kind() {
89        "arithmetic_expr" => Ok(Some(parse_expression(*node, source_code, node, symbols)?)),
90        "null_index" => Ok(None),
91        _ => Err(EssenceParseError::syntax_error(
92            format!("Expected an index, got: '{}'", node.kind()),
93            Some(node.range()),
94        )),
95    }
96}

98fn parse_variable(
99    node: &Node,
100    source_code: &str,
101    symbols: Option<&SymbolTable>,
102) -> Result<Atom, EssenceParseError> {
103    let raw_name = &source_code[node.start_byte()..node.end_byte()];
104    let name = Name::user(raw_name.trim());
105    if let Some(symbols) = symbols {
106        if let Some(decl) = symbols.lookup(&name) {
107            Ok(Atom::Reference(decl))
108        } else {
109            Err(EssenceParseError::syntax_error(
110                format!("Undefined variable: '{raw_name}'"),
111                Some(node.range()),
112            ))
113        }
114    } else {
115        Err(EssenceParseError::syntax_error(
116            format!(
117                "Found variable '{raw_name}'; Did you mean to pass a meta-variable '&{raw_name}'?\n\
118            A symbol table is needed to resolve variable names, but none exists in this context."
119            ),
120            Some(node.range()),
121        ))
122    }
123}

125fn parse_constant(node: &Node, source_code: &str) -> Result<Literal, EssenceParseError> {
126    let inner = named_child!(node);
127    let raw_value = &source_code[inner.start_byte()..inner.end_byte()];
128    match inner.kind() {
129        "integer" => {
130            let value = parse_int(&inner, source_code)?;
131            Ok(Literal::Int(value))
132        }
133        "TRUE" => Ok(Literal::Bool(true)),
134        "FALSE" => Ok(Literal::Bool(false)),
135        _ => Err(EssenceParseError::syntax_error(
136            format!(
137                "'{raw_value}' (kind: '{}') is not a valid constant",
138                inner.kind()
139            ),
140            Some(inner.range()),
141        )),
142    }
143}

145fn parse_int(node: &Node, source_code: &str) -> Result<i32, EssenceParseError> {
146    let raw_value = &source_code[node.start_byte()..node.end_byte()];
147    raw_value.parse::<i32>().map_err(|_e| {
148        if raw_value.is_empty() {
149            EssenceParseError::syntax_error(
150                "Expected an integer here".to_string(),
151                Some(node.range()),
152            )
153        } else {
154            EssenceParseError::syntax_error(
155                format!("'{raw_value}' is not a valid integer"),
156                Some(node.range()),
157            )
158        }
159    })
