5/// Parse an Essence variable domain into its Conjure AST representation.
6pub fn parse_domain(domain: Node, source_code: &str) -> Domain {
7    let domain = domain.child(0).expect("No domain");
8    match domain.kind() {
9        "bool_domain" => Domain::Bool,
10        "int_domain" => parse_int_domain(domain, source_code),
11        "variable" => {
12            let variable_name = &source_code[domain.start_byte()..domain.end_byte()];
13            Domain::Reference(Name::user(variable_name))
14        }
15        _ => panic!("Not bool or int domain"),
16    }
17}

19/// Parse an integer domain. Can be a single integer or a range.
20fn parse_int_domain(int_domain: Node, source_code: &str) -> Domain {
21    if int_domain.child_count() == 1 {
22        Domain::Int(vec![Range::Bounded(i32::MIN, i32::MAX)])
23    } else {
24        let mut ranges: Vec<Range<i32>> = Vec::new();
25        let range_list = int_domain
26            .named_child(0)
27            .expect("No range list found (expression ranges not supported yet");
28        for int_range in named_children(&range_list) {
29            match int_range.kind() {
30                "integer" => {
31                    let integer_value = &source_code[int_range.start_byte()..int_range.end_byte()]
32                        .parse::<i32>()
33                        .unwrap();
34                    ranges.push(Range::Single(*integer_value));
35                }
36                "int_range" => {
37                    let lower_bound: Option<i32>;
38                    let upper_bound: Option<i32>;
39                    let range_component = int_range.child(0).expect("Error with integer range");
40                    match range_component.kind() {
41                        "expression" => {
42                            lower_bound = Some(
43                                source_code
44                                    [range_component.start_byte()..range_component.end_byte()]
45                                    .parse::<i32>()
46                                    .unwrap(),
47                            );
48
49                            if let Some(range_component) = range_component.next_named_sibling() {
50                                upper_bound = Some(
51                                    source_code
52                                        [range_component.start_byte()..range_component.end_byte()]
53                                        .parse::<i32>()
54                                        .unwrap(),
55                                );
56                            } else {
57                                upper_bound = None;
58                            }
59                        }
60                        ".." => {
61                            lower_bound = None;
62                            let range_component = range_component
63                                .next_sibling()
64                                .expect("Error with integer range");
65                            upper_bound = Some(
66                                source_code
67                                    [range_component.start_byte()..range_component.end_byte()]
68                                    .parse::<i32>()
69                                    .unwrap(),
70                            );
71                        }
72                        _ => panic!("unsupported int range type"),
73                    }
74
75                    match (lower_bound, upper_bound) {
76                        (Some(lb), Some(ub)) => ranges.push(Range::Bounded(lb, ub)),
77                        (Some(lb), None) => ranges.push(Range::Bounded(lb, i32::MAX)),
78                        (None, Some(ub)) => ranges.push(Range::Bounded(i32::MIN, ub)),
79                        _ => panic!("Unsupported int range type"),
80                    }
81                }
82                _ => panic!("unsupported int range type"),
83            }
84        }
85        Domain::Int(ranges)
86    }
87}
