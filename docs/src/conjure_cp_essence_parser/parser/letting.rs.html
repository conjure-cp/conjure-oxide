13/// Parse a letting statement into a SymbolTable containing the declared symbols
14pub fn parse_letting_statement(
15    letting_statement: Node,
16    source_code: &str,
17    existing_symbols: Option<&SymbolTable>,
18) -> Result<SymbolTable, EssenceParseError> {
19    let mut symbol_table = SymbolTable::new();
20
21    let mut temp_symbols = BTreeSet::new();
22
23    let variable_list = letting_statement
24        .child_by_field_name("variable_list")
25        .expect("No variable list found");
26    for variable in named_children(&variable_list) {
27        let variable_name = &source_code[variable.start_byte()..variable.end_byte()];
28        temp_symbols.insert(variable_name);
29    }
30
31    let expr_or_domain = letting_statement
32        .child_by_field_name("expr_or_domain")
33        .expect("No domain or expression found for letting statement");
34    match expr_or_domain.kind() {
35        "bool_expr" | "arithmetic_expr" => {
36            for name in temp_symbols {
37                symbol_table.insert(DeclarationPtr::new_value_letting(
38                    Name::user(name),
39                    parse_expression(
40                        expr_or_domain,
41                        source_code,
42                        &letting_statement,
43                        existing_symbols,
44                    )?,
45                ));
46            }
47        }
48        "domain" => {
49            for name in temp_symbols {
50                let domain = parse_domain(expr_or_domain, source_code)?;
51
52                // If it's a record domain, add the field names to the symbol table
53                if let conjure_cp_core::ast::Domain::Record(ref entries) = domain {
54                    for entry in entries {
55                        // Add each field name as a record field declaration
56                        symbol_table.insert(DeclarationPtr::new_record_field(entry.clone()));
57                    }
58                }
59
60                symbol_table.insert(DeclarationPtr::new_domain_letting(Name::user(name), domain));
61            }
62        }
63        _ => {
64            return Err(EssenceParseError::syntax_error(
65                format!(
66                    "Expected letting expression, got '{}'",
67                    expr_or_domain.kind()
68                ),
69                Some(expr_or_domain.range()),
70            ));
71        }
72    }
73
74    Ok(symbol_table)
75}
